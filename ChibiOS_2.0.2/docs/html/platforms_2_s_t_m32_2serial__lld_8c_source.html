00088 /**
00089  * @brief   USART initialization.
00090  * @details This function must be invoked with interrupts disabled.
00091  *
00092  * @param[in] sdp       pointer to a @p SerialDriver object
00093  * @param[in] config    the architecture-dependent serial driver configuration
00094  */
00095 static void usart_init(SerialDriver *sdp, const SerialConfig *config) {
00096   USART_TypeDef *u = sdp->usart;
00097 
00098   /*
00099    * Baud rate setting.
00100    */
00101   if (sdp->usart == USART1)
00102     u->BRR = STM32_PCLK2 / config->sc_speed;
00103   else
00104     u->BRR = STM32_PCLK1 / config->sc_speed;
00105 
00106   /*
00107    * Note that some bits are enforced.
00108    */
00109   u->CR1 = config->sc_cr1 | USART_CR1_UE | USART_CR1_PEIE |
00110                             USART_CR1_RXNEIE | USART_CR1_TE |
00111                             USART_CR1_RE;
00112   u->CR2 = config->sc_cr2 | USART_CR2_LBDIE;
00113   u->CR3 = config->sc_cr3 | USART_CR3_EIE;
00114   (void)u->SR;  /* SR reset step 1.*/
00115   (void)u->DR;  /* SR reset step 2.*/
00116 }

00131 /**
00132  * @brief Error handling routine.
00133  *
00134  * @param[in] sdp       pointer to a @p SerialDriver object
00135  * @param[in] sr        USART SR register value
00136  */
00137 static void set_error(SerialDriver *sdp, uint16_t sr) {
00138   sdflags_t sts = 0;
00139 
00140   if (sr & USART_SR_ORE)
00141     sts |= SD_OVERRUN_ERROR;
00142   if (sr & USART_SR_PE)
00143     sts |= SD_PARITY_ERROR;
00144   if (sr & USART_SR_FE)
00145     sts |= SD_FRAMING_ERROR;
00146   if (sr & USART_SR_NE)
00147     sts |= SD_NOISE_ERROR;
00148   if (sr & USART_SR_LBD)
00149     sts |= SD_BREAK_DETECTED;
00150   chSysLockFromIsr();
00151   sdAddFlagsI(sdp, sts);
00152   chSysUnlockFromIsr();
00153 }

00155 /**
00156  * @brief   Common IRQ handler.
00157  *
00158  * @param[in] sdp       communication channel associated to the USART
00159  */
00160 static void serve_interrupt(SerialDriver *sdp) {
00161   USART_TypeDef *u = sdp->usart;
00162   uint16_t cr1 = u->CR1;
00163   uint16_t sr = u->SR;  /* SR reset step 1.*/
00164   uint16_t dr = u->DR;  /* SR reset step 2.*/
00165 
00166   if (sr & (USART_SR_LBD | USART_SR_ORE | USART_SR_NE |
00167             USART_SR_FE  | USART_SR_PE)) {
00168     set_error(sdp, sr);
00169     u->SR = 0;    /* Clears the LBD bit in the SR.*/
00170   }
00171   if (sr & USART_SR_RXNE) {
00172     chSysLockFromIsr();
00173     sdIncomingDataI(sdp, (uint8_t)dr);
00174     chSysUnlockFromIsr();
00175   }
00176   if ((cr1 & USART_CR1_TXEIE) && (sr & USART_SR_TXE)) {
00177     msg_t b;
00178     chSysLockFromIsr();
00179     b = chOQGetI(&sdp->oqueue);
00180     if (b < Q_OK) {
00181       chEvtBroadcastI(&sdp->oevent);
00182       u->CR1 = cr1 & ~USART_CR1_TXEIE;
00183     }
00184     else
00185       u->DR = b;
00186     chSysUnlockFromIsr();
00187   }
00188 }
