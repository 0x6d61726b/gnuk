00083 /**
00084  * @brief Stops all channels.
00085  *
00086  * @param[in] pwmp pointer to a @p PWMDriver object
00087  */
00088 static void stop_channels(PWMDriver *pwmp) {
00089 
00090   pwmp->pd_enabled_channels = 0;            /* All channels disabled.       */
00091   pwmp->pd_tim->CCER = 0;                   /* Outputs disabled.            */
00092   pwmp->pd_tim->CCR1 = 0;                   /* Comparator 1 disabled.       */
00093   pwmp->pd_tim->CCR2 = 0;                   /* Comparator 2 disabled.       */
00094   pwmp->pd_tim->CCR3 = 0;                   /* Comparator 3 disabled.       */
00095   pwmp->pd_tim->CCR4 = 0;                   /* Comparator 4 disabled.       */
00096   pwmp->pd_tim->CCMR1 = 0;                  /* Channels 1 and 2 frozen.     */
00097   pwmp->pd_tim->CCMR2 = 0;                  /* Channels 3 and 4 frozen.     */
00098 }

00101 /**
00102  * @brief Common TIM2...TIM4 IRQ handler.
00103  * @note It is assumed that the various sources are only activated if the
00104  *       associated callback pointer is not equal to @p NULL in order to not
00105  *       perform an extra check in a potentially critical interrupt handler.
00106  */
00107 static void serve_interrupt(PWMDriver *pwmp) {
00108   uint16_t sr;
00109 
00110   sr = pwmp->pd_tim->SR & pwmp->pd_tim->DIER;
00111   pwmp->pd_tim->SR = ~(TIM_SR_CC1IF | TIM_SR_CC2IF | TIM_SR_CC3IF |
00112                        TIM_SR_CC4IF | TIM_SR_UIF);
00113   if ((sr & TIM_SR_CC1IF) != 0)
00114     pwmp->pd_config->pc_channels[0].pcc_callback();
00115   if ((sr & TIM_SR_CC2IF) != 0)
00116     pwmp->pd_config->pc_channels[1].pcc_callback();
00117   if ((sr & TIM_SR_CC3IF) != 0)
00118     pwmp->pd_config->pc_channels[2].pcc_callback();
00119   if ((sr & TIM_SR_CC4IF) != 0)
00120     pwmp->pd_config->pc_channels[3].pcc_callback();
00121   if ((sr & TIM_SR_UIF) != 0)
00122     pwmp->pd_config->pc_callback();
00123 }

00269 /**
00270  * @brief Configures and activates the PWM peripheral.
00271  *
00272  * @param[in] pwmp pointer to a @p PWMDriver object
00273  */
00274 void pwm_lld_start(PWMDriver *pwmp) {
00275   uint16_t ccer;
00276 
00277   if (pwmp->pd_state == PWM_STOP) {
00278     /* Clock activation.*/
00279 #if USE_STM32_PWM1
00280     if (&PWMD1 == pwmp) {
00281       NVICEnableVector(TIM1_UP_IRQn,
00282                        CORTEX_PRIORITY_MASK(STM32_PWM1_IRQ_PRIORITY));
00283       NVICEnableVector(TIM1_CC_IRQn,
00284                        CORTEX_PRIORITY_MASK(STM32_PWM1_IRQ_PRIORITY));
00285       RCC->APB2ENR |= RCC_APB2ENR_TIM1EN;
00286     }
00287 #endif
00288 #if USE_STM32_PWM2
00289     if (&PWMD2 == pwmp) {
00290       NVICEnableVector(TIM2_IRQn,
00291                        CORTEX_PRIORITY_MASK(STM32_PWM2_IRQ_PRIORITY));
00292       RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
00293     }
00294 #endif
00295 #if USE_STM32_PWM3
00296     if (&PWMD3 == pwmp) {
00297       NVICEnableVector(TIM3_IRQn,
00298                        CORTEX_PRIORITY_MASK(STM32_PWM3_IRQ_PRIORITY));
00299       RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
00300     }
00301 #endif
00302 #if USE_STM32_PWM4
00303     if (&PWMD4 == pwmp) {
00304       NVICEnableVector(TIM4_IRQn,
00305                        CORTEX_PRIORITY_MASK(STM32_PWM4_IRQ_PRIORITY));
00306       RCC->APB1ENR |= RCC_APB1ENR_TIM4EN;
00307     }
00308 #endif
00309   }
00310   /* Reset channels.*/
00311   stop_channels(pwmp);
00312 
00313   /* Configuration or reconfiguration.*/
00314   pwmp->pd_tim->CR1  = 0;                   /* Timer stopped.               */
00315   pwmp->pd_tim->SMCR = 0;                   /* Slave mode disabled.         */
00316   pwmp->pd_tim->CR2  = pwmp->pd_config->pc_cr2;
00317   pwmp->pd_tim->PSC  = pwmp->pd_config->pc_psc;
00318   pwmp->pd_tim->CNT  = 0;
00319   pwmp->pd_tim->ARR  = pwmp->pd_config->pc_arr;
00320   /* Output enables and polarities setup.*/
00321   ccer = 0;
00322   switch (pwmp->pd_config->pc_channels[0].pcc_mode) {
00323   case PWM_OUTPUT_ACTIVE_LOW:
00324     ccer |= TIM_CCER_CC1P;
00325   case PWM_OUTPUT_ACTIVE_HIGH:
00326     ccer |= TIM_CCER_CC1E;
00327   default:
00328     ;
00329   }
00330   switch (pwmp->pd_config->pc_channels[1].pcc_mode) {
00331   case PWM_OUTPUT_ACTIVE_LOW:
00332     ccer |= TIM_CCER_CC2P;
00333   case PWM_OUTPUT_ACTIVE_HIGH:
00334     ccer |= TIM_CCER_CC2E;
00335   default:
00336     ;
00337   }
00338   switch (pwmp->pd_config->pc_channels[2].pcc_mode) {
00339   case PWM_OUTPUT_ACTIVE_LOW:
00340     ccer |= TIM_CCER_CC3P;
00341   case PWM_OUTPUT_ACTIVE_HIGH:
00342     ccer |= TIM_CCER_CC3E;
00343   default:
00344     ;
00345   }
00346   switch (pwmp->pd_config->pc_channels[3].pcc_mode) {
00347   case PWM_OUTPUT_ACTIVE_LOW:
00348     ccer |= TIM_CCER_CC4P;
00349   case PWM_OUTPUT_ACTIVE_HIGH:
00350     ccer |= TIM_CCER_CC4E;
00351   default:
00352     ;
00353   }
00354   pwmp->pd_tim->CCER = ccer;
00355   pwmp->pd_tim->EGR  = TIM_EGR_UG;          /* Update event.                */
00356   pwmp->pd_tim->SR   = 0;                   /* Clear pending IRQs.          */
00357   pwmp->pd_tim->DIER = pwmp->pd_config->pc_callback == NULL ? 0 : TIM_DIER_UIE;
00358   pwmp->pd_tim->BDTR = TIM_BDTR_MOE;
00359   pwmp->pd_tim->CR1  = TIM_CR1_ARPE | TIM_CR1_URS |
00360                        TIM_CR1_CEN;         /* Timer configured and started.*/
00361 }

00404 /**
00405  * @brief Enables a PWM channel.
00406  *
00407  * @param[in] pwmp      pointer to a @p PWMDriver object
00408  * @param[in] channel   PWM channel identifier
00409  * @param[in] width     PWM pulse width as clock pulses number
00410  */
00411 void pwm_lld_enable_channel(PWMDriver *pwmp,
00412                             pwmchannel_t channel,
00413                             pwmcnt_t width) {
00414 
00415   /*
00416    * Changes the pulse width.
00417    */
00418   switch (channel) {
00419   case 0:
00420     pwmp->pd_tim->CCR1 = width;
00421     break;
00422   case 1:
00423     pwmp->pd_tim->CCR2 = width;
00424     break;
00425   case 2:
00426     pwmp->pd_tim->CCR3 = width;
00427     break;
00428   case 3:
00429     pwmp->pd_tim->CCR4 = width;
00430     break;
00431   }
00432   if ((pwmp->pd_enabled_channels & (1 << channel)) == 0) {
00433     /*
00434      * The channel is not enabled yet.
00435      */
00436     pwmp->pd_enabled_channels |= (1 << channel);
00437     /*
00438      * Setup the comparator, the channel is configured as PWM mode 1 with
00439      * preload enabled.
00440      */
00441     switch (channel) {
00442     case 0:
00443       pwmp->pd_tim->CCMR1 = (pwmp->pd_tim->CCMR1 & 0xFF00) |
00444                             TIM_CCMR1_OC1M_1 | TIM_CCMR1_OC1M_2 |
00445                             TIM_CCMR1_OC1PE;
00446       pwmp->pd_tim->SR = ~TIM_SR_CC1IF;
00447       pwmp->pd_tim->DIER |= pwmp->pd_config->pc_channels[0].pcc_callback == NULL
00448                             ? 0 : TIM_DIER_CC1IE;
00449       break;
00450     case 1:
00451       pwmp->pd_tim->CCMR1 = (pwmp->pd_tim->CCMR1 & 0x00FF) |
00452                             TIM_CCMR1_OC2M_1 | TIM_CCMR1_OC2M_2 |
00453                             TIM_CCMR1_OC2PE;
00454       pwmp->pd_tim->SR = ~TIM_SR_CC2IF;
00455       pwmp->pd_tim->DIER |= pwmp->pd_config->pc_channels[1].pcc_callback == NULL
00456                             ? 0 : TIM_DIER_CC2IE;
00457       break;
00458     case 2:
00459       pwmp->pd_tim->CCMR2 = (pwmp->pd_tim->CCMR2 & 0xFF00) |
00460                             TIM_CCMR2_OC3M_1 | TIM_CCMR2_OC3M_2 |
00461                             TIM_CCMR2_OC3PE;
00462       pwmp->pd_tim->SR = ~TIM_SR_CC3IF;
00463       pwmp->pd_tim->DIER |= pwmp->pd_config->pc_channels[2].pcc_callback == NULL
00464                             ? 0 : TIM_DIER_CC3IE;
00465       break;
00466     case 3:
00467       pwmp->pd_tim->CCMR2 = (pwmp->pd_tim->CCMR2 & 0x00FF) |
00468                             TIM_CCMR2_OC4M_1 | TIM_CCMR2_OC4M_2 |

00470       pwmp->pd_tim->SR = ~TIM_SR_CC4IF;

00472                             ? 0 : TIM_DIER_CC4IE;
