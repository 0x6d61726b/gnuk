00103 /**
00104  * @brief   Virtual timers ticker.
00105  */
00106 #define chVTDoTickI() {                                                 \
00107   vtlist.vt_systime++;                                                  \
00108   if (&vtlist != (VTList *)vtlist.vt_next) {                            \
00109     VirtualTimer *vtp;                                                  \
00110                                                                         \
00111     --vtlist.vt_next->vt_time;                                          \
00112     while (!(vtp = vtlist.vt_next)->vt_time) {                          \
00113       vtfunc_t fn = vtp->vt_func;                                       \
00114       vtp->vt_func = (vtfunc_t)NULL;                                    \
00115       vtp->vt_next->vt_prev = (void *)&vtlist;                          \
00116       (&vtlist)->vt_next = vtp->vt_next;                                \
00117       fn(vtp->vt_par);                                                  \
00118     }                                                                   \
00119   }                                                                     \
00120 }
