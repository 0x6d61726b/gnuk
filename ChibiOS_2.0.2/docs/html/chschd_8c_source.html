00065 /**
00066  * @brief   Inserts a thread in the Ready List.
00067  * @note    The function does not reschedule, the @p chSchRescheduleS() should
00068  *          be called soon after.
00069  *
00070  * @param[in] tp        the Thread to be made ready
00071  * @return              The Thread pointer.
00072  */
00073 #if !defined(PORT_OPTIMIZED_READYI) || defined(__DOXYGEN__)
00074 #if CH_OPTIMIZE_SPEED
00075 /* NOTE: it is inlined in this module only.*/
00076 INLINE Thread *chSchReadyI(Thread *tp) {
00077 #else
00078 Thread *chSchReadyI(Thread *tp) {
00079 #endif
00080   Thread *cp;
00081 
00082   tp->p_state = THD_STATE_READY;
00083   cp = (Thread *)&rlist.r_queue;
00084   do {
00085     cp = cp->p_next;
00086   } while (cp->p_prio >= tp->p_prio);
00087   /* Insertion on p_prev.*/
00088   tp->p_prev = (tp->p_next = cp)->p_prev;
00089   tp->p_prev->p_next = cp->p_prev = tp;
00090   return tp;
00091 }

00143  * @brief   Puts the current thread to sleep into the specified state with
00144  *          timeout specification.
00145  * @details The thread goes into a sleeping state, if it is not awakened
00146  *          explicitly within the specified timeout then it is forcibly
00147  *          awakened with a @p RDY_TIMEOUT low level message. The @ref
00148  *          thread_states are described into @p threads.h.
00149  *
00150  * @param[in] newstate  the new thread state

00155  *                        @p chSchGoSleepS() but, of course, less efficient.

00157  *                        interpreted as a normal time specification not as an

00159  *                      .
00160  * @return              The wakeup message.
00161  * @retval RDY_TIMEOUT if a timeout occurs.
00162  */
00163 msg_t chSchGoSleepTimeoutS(tstate_t newstate, systime_t time) {
00164 
00165   if (TIME_INFINITE != time) {
00166     VirtualTimer vt;
00167 
00168     chVTSetI(&vt, time, wakeup, currp);
00169     chSchGoSleepS(newstate);
00170     if (chVTIsArmedI(&vt))
00171       chVTResetI(&vt);
00172   }
00173   else

00179 /**
00180  * @brief   Wakes up a thread.
00181  * @details The thread is inserted into the ready list or immediately made
00182  *          running depending on its relative priority compared to the current
00183  *          thread.
00184  * @note    It is equivalent to a @p chSchReadyI() followed by a
00185  *          @p chSchRescheduleS() but much more efficient.
00186  * @note    The function assumes that the current thread has the highest
00187  *          priority.
00188  *
00189  * @param[in] ntp       the Thread to be made ready
00190  * @param[in] msg       message to the awakened thread
00191  */
00192 #if !defined(PORT_OPTIMIZED_WAKEUPS) || defined(__DOXYGEN__)
00193 void chSchWakeupS(Thread *ntp, msg_t msg) {
00194 
00195   ntp->p_u.rdymsg = msg;
00196   /* If the waken thread has a not-greater priority than the current
00197      one then it is just inserted in the ready list else it made
00198      running immediately and the invoking thread goes in the ready
00199      list instead.*/
00200   if (ntp->p_prio <= currp->p_prio)
00201     chSchReadyI(ntp);
00202   else {
00203     Thread *otp = chSchReadyI(currp);
00204 #if CH_TIME_QUANTUM > 0
00205     rlist.r_preempt = CH_TIME_QUANTUM;
00206 #endif
00207     setcurrp(ntp);
00208     ntp->p_state = THD_STATE_CURRENT;
00209     chDbgTrace(otp);
00210     chSysSwitchI(ntp, otp);
00211   }
00212 }
00213 #endif /* !defined(PORT_OPTIMIZED_WAKEUPS) */

00250 /**
00251  * @brief   Evaluates if a reschedule is required.
00252  * @details The decision is taken by comparing the relative priorities and
00253  *          depending on the state of the round robin timeout counter.
00254  * @note    This function is meant to be used in the timer interrupt handler
00255  *          where @p chVTDoTickI() is invoked.
00256  *
00257  * @retval TRUE         if there is a thread that should go in running state.
00258  * @retval FALSE        if a reschedule is not required.
00259  */
00260 #if !defined(PORT_OPTIMIZED_ISRESCHREQUIREDEXI) || defined(__DOXYGEN__)
00261 bool_t chSchIsRescRequiredExI(void) {
00262   tprio_t p1 = firstprio(&rlist.r_queue);
00263   tprio_t p2 = currp->p_prio;
00264 #if CH_TIME_QUANTUM > 0
00265   /* If the running thread has not reached its time quantum, reschedule only
00266      if the first thread on the ready queue has a higher priority.
00267      Otherwise, if the running thread has used up its time quantum, reschedule
00268      if the first thread on the ready queue has equal or higher priority.*/
00269   return rlist.r_preempt ? p1 > p2 : p1 >= p2;
00270 #else
00271   /* If the round robin preemption feature is not enabled then performs a
00272      simpler comparison.*/
00273   return p1 > p2;
00274 #endif
00275 }
00276 #endif /* !defined(PORT_OPTIMIZED_ISRESCHREQUIREDEXI) */
