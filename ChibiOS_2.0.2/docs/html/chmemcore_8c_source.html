00027 /**
00028  * @file    chmemcore.c
00029  * @brief   Core memory manager code.
00030  *
00031  * @addtogroup memcore
00032  * @details Core Memory Manager related APIs and services.
00033  *          <h2>Operation mode</h2>
00034  *          The core memory manager is a simplified allocator that only allows
00035  *          to allocate memory blocks without the possibility to free them.<br>
00036  *          This allocator is meant as a memory blocks provider for the other
00037  *          allocators such as:
00038  *          - C-Runtime allocator (through a compiler specific adapter module).
00039  *          - Heap allocator (see @ref heaps).
00040  *          - Memory pools allocator (see @ref pools).
00041  *          .
00042  *          By having a centralized memory provider the various allocators can
00043  *          coexist and share the main memory.<br>
00044  *          This allocator, alone, is also useful for very simple applications
00045  *          that just require a simple way to get memory blocks.<br>
00046  *          In order to use the core memory manager APIs the @p CH_USE_MEMCORE
00047  *          option must be enabled in @p chconf.h.
00048  * @{
00049  */

00060  * @note    Internal use only.

00062 void core_init(void) {
00063 #if CH_MEMCORE_SIZE == 0
00064   extern uint8_t __heap_base__;
00065   extern uint8_t __heap_end__;
00066   nextmem = &__heap_base__;
00067   endmem = &__heap_end__;

00077 /**
00078  * @brief   Allocates a memory block.
00079  * @details The size of the returned block is aligned to the alignment
00080  *          type @p stkalign_t so it is not possible to allocate less
00081  *          than <code>sizeof(stkalign_t)</code>.
00082  *
00083  *
00084  * @param[in] size      the size of the block to be allocated
00085  * @return              A pointer to the allocated memory block.
00086  * @retval NULL         allocation failed, core memory exhausted.
00087  */
00088 void *chCoreAlloc(size_t size) {
00089   void *p;
00090 
00091   chSysLock();
00092   p = chCoreAllocI(size);
00093   chSysUnlock();
00094   return p;
00095 }

00097 /**
00098  * @brief   Allocates a memory block.
00099  * @details The size of the returned block is aligned to the alignment
00100  *          type @p align_t so it is not possible to allocate less than
00101  *          <code>sizeof(align_t)</code>.
00102  *
00103  * @param[in] size      the size of the block to be allocated.
00104  * @return              A pointer to the allocated memory block.
00105  * @retval NULL         allocation failed, core memory exhausted.
00106  */
00107 void *chCoreAllocI(size_t size) {
00108   void *p;
00109 
00110   size = MEM_ALIGN_SIZE(size);
00111   if ((size_t)(endmem - nextmem) < size)
00112     return NULL;
00113   p = nextmem;
00114   nextmem += size;
00115   return p;
00116 }
