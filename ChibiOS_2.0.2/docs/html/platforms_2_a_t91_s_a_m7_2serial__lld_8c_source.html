00096 /**
00097  * @brief   USART initialization.
00098  *
00099  * @param[in] sdp       communication channel associated to the USART
00100  * @param[in] config    the architecture-dependent serial driver configuration
00101  */
00102 static void usart_init(SerialDriver *sdp, const SerialConfig *config) {
00103   AT91PS_USART u = sdp->usart;
00104 
00105   /* Disables IRQ sources and stop operations.*/
00106   u->US_IDR = 0xFFFFFFFF;
00107   u->US_CR = AT91C_US_RSTRX | AT91C_US_RSTTX | AT91C_US_RSTSTA;
00108 
00109   /* New parameters setup.*/
00110   if (config->sc_mr & AT91C_US_OVER)
00111     u->US_BRGR = MCK / (config->sc_speed * 8);
00112   else
00113     u->US_BRGR = MCK / (config->sc_speed * 16);
00114   u->US_MR = config->sc_mr;
00115   u->US_RTOR = 0;
00116   u->US_TTGR = 0;
00117 
00118   /* Enables operations and IRQ sources.*/
00119   u->US_CR = AT91C_US_RXEN | AT91C_US_TXEN | AT91C_US_DTREN | AT91C_US_RTSEN;
00120   u->US_IER = AT91C_US_RXRDY | AT91C_US_OVRE | AT91C_US_FRAME | AT91C_US_PARE |
00121               AT91C_US_RXBRK;
00122 }

00139 /**
00140  * @brief   Error handling routine.
00141  *
00142  * @param[in] err       USART CSR register value
00143  * @param[in] sdp       communication channel associated to the USART
00144  */
00145 static void set_error(SerialDriver *sdp, AT91_REG csr) {
00146   sdflags_t sts = 0;
00147 
00148   if (csr & AT91C_US_OVRE)
00149     sts |= SD_OVERRUN_ERROR;
00150   if (csr & AT91C_US_PARE)
00151     sts |= SD_PARITY_ERROR;
00152   if (csr & AT91C_US_FRAME)
00153     sts |= SD_FRAMING_ERROR;
00154   if (csr & AT91C_US_RXBRK)
00155     sts |= SD_BREAK_DETECTED;
00156   chSysLockFromIsr();
00157   sdAddFlagsI(sdp, sts);
00158   chSysUnlockFromIsr();

00167 /**
00168  * @brief   Common IRQ handler.
00169  *
00170  * @param[in] sdp       communication channel associated to the USART
00171  */
00172 void sd_lld_serve_interrupt(SerialDriver *sdp) {
00173   uint32_t csr;
00174   AT91PS_USART u = sdp->usart;
00175 
00176   csr = u->US_CSR;
00177   if (csr & AT91C_US_RXRDY) {
00178     chSysLockFromIsr();
00179     sdIncomingDataI(sdp, u->US_RHR);
00180     chSysUnlockFromIsr();
00181   }
00182   if ((u->US_IMR & AT91C_US_TXRDY) && (csr & AT91C_US_TXRDY)) {
00183     msg_t b;
00184 
00185     chSysLockFromIsr();
00186     b = chOQGetI(&sdp->oqueue);
00187     if (b < Q_OK) {
00188       chEvtBroadcastI(&sdp->oevent);
00189       u->US_IDR = AT91C_US_TXRDY;
00190     }
00191     else
00192       u->US_THR = b;
00193     chSysUnlockFromIsr();
00194   }
00195   csr &= (AT91C_US_OVRE | AT91C_US_FRAME | AT91C_US_PARE | AT91C_US_RXBRK);
00196   if (csr != 0) {
00197     set_error(sdp, csr);
00198     u->US_CR = AT91C_US_RSTSTA;
00199   }
00200   AT91C_BASE_AIC->AIC_EOICR = 0;
00201 }

00294 /**
00295  * @brief   Low level serial driver configuration and (re)start.
00296  *
00297  * @param[in] sdp       pointer to a @p SerialDriver object
00298  * @param[in] config    the architecture-dependent serial driver configuration.
00299  *                      If this parameter is set to @p NULL then a default
00300  *                      configuration is used.
00301  */
00302 void sd_lld_start(SerialDriver *sdp, const SerialConfig *config) {
00303 
00304   if (config == NULL)
00305     config = &default_config;
00306 
00307   if (sdp->state == SD_STOP) {
00308 #if USE_SAM7_USART0
00309     if (&SD1 == sdp) {
00310       /* Starts the clock and clears possible sources of immediate interrupts.*/
00311       AT91C_BASE_PMC->PMC_PCER = (1 << AT91C_ID_US0);
00312       /* Enables associated interrupt vector.*/
00313       AIC_EnableIT(AT91C_ID_US0);
00314     }
00315 #endif
00316 #if USE_SAM7_USART1
00317     if (&SD2 == sdp) {
00318       /* Starts the clock and clears possible sources of immediate interrupts.*/
00319       AT91C_BASE_PMC->PMC_PCER = (1 << AT91C_ID_US1);
00320       /* Enables associated interrupt vector.*/
00321       AIC_EnableIT(AT91C_ID_US1);
00322     }
00323 #endif
00324   // note - no explicit start for SD3 (DBGU_UART) since it's not included in the AIC or PMC
00325   }
00326   usart_init(sdp, config);
00327 }

00329 /**
00330  * @brief   Low level serial driver stop.
00331  * @details De-initializes the USART, stops the associated clock, resets the

00333  *
00334  * @param[in] sdp       pointer to a @p SerialDriver object

00336 void sd_lld_stop(SerialDriver *sdp) {
00337

00346 #endif
00347 #if USE_SAM7_USART1
00348     if (&SD2 == sdp) {
00349       AT91C_BASE_PMC->PMC_PCDR = (1 << AT91C_ID_US1);
00350       AIC_DisableIT(AT91C_ID_US1);

00353 #endif
00354 #if USE_SAM7_DBGU_UART
00355     if (&SD3 == sdp) {
00356       AT91C_BASE_DBGU->DBGU_IDR = 0xFFFFFFFF;

00359 #endif
00360   }
