00063 /**
00064  * @brief   Post-IRQ switch code.
00065  * @details On entry the stack and the registers are restored by the exception
00066  *          return, the PC value is stored in @p _port_saved_pc, the interrupts
00067  *          are disabled.
00068  */
00069 #if !defined(__DOXYGEN__)
00070 __attribute__((naked))
00071 #endif
00072 void _port_switch_from_irq(void) {
00073   /* Note, saves r4 to make space for the PC.*/
00074   asm volatile ("push    {r0, r1, r2, r3, r4}                   \n\t"
00075                 "mrs     r0, APSR                               \n\t"
00076                 "mov     r1, r12                                \n\t"
00077                 "push    {r0, r1, lr}                           \n\t"
00078                 "ldr     r0, =_port_saved_pc                    \n\t"
00079                 "ldr     r0, [r0]                               \n\t"
00080                 "add   r0, r0, #1               \n\t"
00081                 "str     r0, [sp, #28]");
00082 
00083   chSchDoRescheduleI();
00084 
00085   /* Note, the last registers are restored alone after re-enabling the
00086      interrupts in order to minimize the (very remote and unlikely)
00087      possibility that the stack is filled by continuous and saturating
00088      interrupts that would not allow that last words to be pulled out of
00089      the stack.*/
00090   asm volatile ("pop     {r0, r1, r2}                           \n\t"
00091                 "mov     r12, r1                                \n\t"
00092                 "msr     APSR, r0                               \n\t"
00093                 "mov     lr, r2                                 \n\t"
00094                 "cpsie   i                                      \n\t"
00095                 "pop     {r0, r1, r2, r3, pc}");
00096 }

00100                 "mov     r4, r8                                 \n\t"       \

00110                 "mov     r9, r5                                 \n\t"       \

00112                 "mov     r11, r7                                \n\t"       \
00113                 "pop     {r4, r5, r6, r7, pc}" :  : "r" (sp));              \
00114 }

00116 /**
00117  * @brief   Performs a context switch between two threads.
00118  * @details This is the most critical code in any port, this function
00119  *          is responsible for the context switch between 2 threads.
00120  * @note    The implementation of this code affects <b>directly</b> the context
00121  *          switch performance so optimize here as much as you can.
00122  *
00123  * @param[in] ntp       the thread to be switched in
00124  * @param[in] otp       the thread to be switched out
00125  */
00126 #if !defined(__DOXYGEN__)
00127 __attribute__((naked))
00128 #endif
00129 void port_switch(Thread *ntp, Thread *otp) {
00130   register struct intctx *r13 asm ("r13");
00131 
00132   /* Stack overflow check, if enabled.*/
00133 #if CH_DBG_ENABLE_STACK_CHECK
00134   if ((void *)(r13 - 1) < (void *)(otp + 1))
00135     asm volatile ("movs    r0, #0                               \n\t"
00136                   "b       chDbgPanic");
00137 #endif /* CH_DBG_ENABLE_STACK_CHECK */
00138 
00139   PUSH_CONTEXT(r13);
00140 
00141   otp->p_ctx.r13 = r13;
00142   r13 = ntp->p_ctx.r13;
00143 
00144   POP_CONTEXT(r13);
00145 }
