00089 /**
00090  * @brief   SVC vector.
00091  * @details The SVC vector is used for exception mode re-entering after a
00092  *          context switch.
00093  */
00094 void SVCallVector(void) {
00095   register struct extctx *ctxp;
00096 
00097   /* Discarding the current exception context and positioning the stack to
00098      point to the real one.*/
00099   asm volatile ("mrs     %0, PSP" : "=r" (ctxp) : );
00100   ctxp++;
00101   asm volatile ("msr     PSP, %0" :  : "r" (ctxp));
00102   port_unlock_from_isr();

00105 /**
00106  * @brief   Reschedule verification and setup after an IRQ.
00107  */
00108 void _port_irq_epilogue(void) {
00109 
00110   port_lock_from_isr();
00111   if ((SCB_ICSR & ICSR_RETTOBASE) && chSchIsRescRequiredExI()) {
00112     register struct extctx *ctxp;
00113 
00114     /* Adding an artificial exception return context, there is no need to
00115        populate it fully.*/
00116     asm volatile ("mrs     %0, PSP" : "=r" (ctxp) : );
00117     ctxp--;
00118     asm volatile ("msr     PSP, %0" :  : "r" (ctxp));
00119     ctxp->pc = _port_switch_from_isr;
00120     ctxp->xpsr = (regarm_t)0x01000000;
00121     /* Note, returning without unlocking is intentional, this is done in
00122        order to keep the rest of the context switching atomic.*/
00123     return;
00124   }
00125   /* ISR exit without context switching.*/
00126   port_unlock_from_isr();
00127 }

00143  * @brief   Performs a context switch between two threads.
00144  * @details This is the most critical code in any port, this function
00145  *          is responsible for the context switch between 2 threads.
00146  * @note    The implementation of this code affects <b>directly</b> the context
00147  *          switch performance so optimize here as much as you can.
00148  *
00149  * @param[in] ntp       the thread to be switched in
00150  * @param[in] otp       the thread to be switched out

00152 #if !defined(__DOXYGEN__)
00153 __attribute__((naked))
00154 #endif
00155 void port_switch(Thread *ntp, Thread *otp) {
00156 
00157 #if CH_DBG_ENABLE_STACK_CHECK
00158   /* Stack overflow check, if enabled.*/
00159   register struct intctx *r13 asm ("r13");
00160   if ((void *)(r13 - 1) < (void *)(otp + 1))
00161     asm volatile ("movs    r0, #0                               \n\t"
00162                   "b       chDbgPanic");
00163 #endif /* CH_DBG_ENABLE_STACK_CHECK */
00164 
00165   PUSH_CONTEXT();
00166 
00167   asm volatile ("str     sp, [%1, #12]                          \n\t"
00168                 "ldr     sp, [%0, #12]" : : "r" (ntp), "r" (otp));
00169 
00170   POP_CONTEXT();
00171 }
