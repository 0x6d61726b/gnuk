00058 /**
00059  * @brief   Low level HAL driver initialization.
00060  */
00061 void hal_lld_init(void) {
00062   extern void _vectors(void);
00063   uint32_t n;
00064 
00065   /* Enables the branch prediction, clears and enables the BTB into the
00066      BUCSR special register (1013).*/
00067   asm volatile ("li      %%r3, 0x0201          \t\n"
00068                 "mtspr   1013, %%r3": : : "r3");
00069 
00070   /* FLASH wait states and prefetching setup.*/
00071   CFLASH0.BIUCR.R  = SPC563_FLASH_BIUCR | SPC563_FLASH_WS;
00072   CFLASH0.BIUCR2.R = 0;
00073   CFLASH0.PFCR3.R  = 0;
00074 
00075   /* Optimal crossbar settings. The DMA priority is placed above the CPU
00076      priority in order to not starve I/O activities while the CPU is
00077      excuting tight loops (FLASH and SRAM slave ports only).
00078      The SRAM is parked on the load/store port, for some unknown reason it
00079      is defaulted on the instructions port and this kills performance.*/
00080   XBAR.SGPCR3.B.PARK = 4;               /* RAM slave on load/store port.*/
00081   XBAR.MPR0.R = 0x00030201;             /* Flash slave port priorities:
00082                                             eDMA (1):              0 (highest)
00083                                             Core Instructions (0): 1
00084                                             Undocumented (2):      2
00085                                             Core Data (4):         3        */
00086   XBAR.MPR3.R = 0x00030201;             /* SRAM slave port priorities:
00087                                             eDMA (1):              0 (highest)
00088                                             Core Instructions (0): 1
00089                                             Undocumented (2):      2
00090                                             Core Data (4):         3        */
00091 
00092   /* Downcounter timer initialized for system tick use, TB enabled for debug
00093      and measurements.*/
00094   n = SPC563_SYSCLK / CH_FREQUENCY;
00095   asm volatile ("li      %%r3, 0            \t\n"
00096                 "mtspr   284, %%r3          \t\n"   /* Clear TBL register.  */
00097                 "mtspr   285, %%r3          \t\n"   /* Clear TBU register.  */
00098                 "mtspr   22, %[n]           \t\n"   /* Init. DEC register.  */
00099                 "mtspr   54, %[n]           \t\n"   /* Init. DECAR register.*/
00100                 "li      %%r3, 0x4000       \t\n"   /* TBEN bit.            */
00101                 "mtspr   1008, %%r3         \t\n"   /* HID0 register.       */
00102                 "lis     %%r3, 0x0440       \t\n"   /* DIE ARE bits.        */
00103                 "mtspr   340, %%r3"                 /* TCR register.        */
00104                 : : [n] "r" (n) : "r3");
00105 
00106   /* INTC initialization, software vector mode, 4 bytes vectors, starting
00107      at priority 0.*/
00108   INTC.MCR.R   = 0;
00109   INTC.CPR.R   = 0;
00110   INTC.IACKR.R = (uint32_t)_vectors;
00111 }

00113 /**
00114  * @brief   SPC563 clocks and PLL initialization.
00115  * @note    All the involved constants come from the file @p board.h and
00116  *          @p hal_lld.h
00117  */
00118 void spc563_clock_init(void) {
00119 
00120   /* PLL activation.*/
00121   FMPLL.ESYNCR1.B.EMODE     = 1;
00122   FMPLL.ESYNCR1.B.CLKCFG   &= 1;                    /* Bypass mode, PLL off.*/
00123   FMPLL.ESYNCR1.B.CLKCFG   |= 2;                    /* PLL on.              */
00124   FMPLL.ESYNCR1.B.EPREDIV   = SPC563_CLK_PREDIV;
00125   FMPLL.ESYNCR1.B.EMFD      = SPC563_CLK_MFD;
00126   FMPLL.ESYNCR2.B.ERFD      = SPC563_CLK_RFD;
00127   while (!FMPLL.SYNSR.B.LOCK)
00128     ;
00129   FMPLL.ESYNCR1.B.CLKCFG   |= 4;                    /* Clock from the PLL.  */
00130 }
