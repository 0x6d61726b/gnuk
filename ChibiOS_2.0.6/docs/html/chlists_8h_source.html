00053 /**
00054  * @brief   Evaluates to @p TRUE if the specified threads queue or list is
00055  * empty.
00056  */
00057 #define isempty(p)      ((p)->p_next == (Thread *)(p))

00059 /**
00060  * @brief   Evaluates to @p TRUE if the specified threads queue or list is

00063 #define notempty(p)     ((p)->p_next != (Thread *)(p))
