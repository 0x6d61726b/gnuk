00074 /**
00075  * @brief   eSCI initialization.
00076  * @details This function must be invoked with interrupts disabled.
00077  *
00078  * @param[in] sdp       pointer to a @p SerialDriver object
00079  * @param[in] config    the architecture-dependent serial driver configuration
00080  */
00081 static void esci_init(SerialDriver *sdp, const SerialConfig *config) {
00082   volatile struct ESCI_tag *escip = sdp->escip;
00083   uint8_t mode = config->sc_mode;
00084 
00085   escip->CR2.R  = 0;                /* MDIS off.                            */
00086   escip->CR1.R  = 0;
00087   escip->LCR.R  = 0;
00088   escip->CR1.B.SBR = SPC563_SYSCLK / (16 * config->sc_speed);
00089   if (mode & SD_MODE_LOOPBACK)
00090     escip->CR1.B.LOOPS = 1;
00091   switch (mode & SD_MODE_PARITY) {
00092   case SD_MODE_PARITY_ODD:
00093     escip->CR1.B.PT = 1;
00094   case SD_MODE_PARITY_EVEN:
00095     escip->CR1.B.PE = 1;
00096     escip->CR1.B.M  = 1;            /* Makes it 8 bits data + 1 bit parity. */
00097   default:
00098     ;
00099   }
00100   escip->LPR.R  = 0;
00101   escip->CR1.R |= 0x0000002C;       /* RIE, TE, RE to 1.                    */
00102   escip->CR2.R |= 0x000F;           /* ORIE, NFIE, FEIE, PFIE to 1.         */
00103 }

00125 static void set_error(SerialDriver *sdp, uint32_t sr) {
00126   sdflags_t sts = 0;

00128   if (sr & 0x08000000)

00143 /**
00144  * @brief   Common IRQ handler.
00145  *
00146  * @param[in] sdp       pointer to a @p SerialDriver object
00147  */
00148 static void serve_interrupt(SerialDriver *sdp) {
00149   volatile struct ESCI_tag *escip = sdp->escip;
00150 
00151   uint32_t sr = escip->SR.R;
00152   escip->SR.R = 0x3FFFFFFF;                     /* Does not clear TDRE | TC.*/
00153   if (sr & 0x0F000000)                          /* OR | NF | FE | PF.       */
00154     set_error(sdp, sr);
00155   if (sr & 0x20000000) {                        /* RDRF.                    */
00156     chSysLockFromIsr();
00157     sdIncomingDataI(sdp, escip->DR.B.D);
00158     chSysUnlockFromIsr();
00159   }
00160   if (escip->CR1.B.TIE && (sr & 0x80000000)) {  /* TDRE.                    */
00161     msg_t b;
00162     chSysLockFromIsr();
00163     b = chOQGetI(&sdp->oqueue);
00164     if (b < Q_OK) {
00165       chEvtBroadcastI(&sdp->oevent);
00166       escip->CR1.B.TIE = 0;
00167     }
00168     else {
00169       ESCI_A.SR.B.TDRE = 1;
00170       escip->DR.R = (uint16_t)b;
00171     }
00172     chSysUnlockFromIsr();
00173   }
00174 }
