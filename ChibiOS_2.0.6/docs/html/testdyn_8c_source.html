00243   tp = chThdCreateFromHeap(&heap1, WA_SIZE, prio-1, thread, "A");

00249 
00250   /* Verify the new threads count.*/
00251   n2 = regscan();
00252   test_assert(4, n1 == n2 - 1, "unexpected threads count");
00253 
00254   /* Detach and let the thread execute and terminate.*/
00255   chThdRelease(tp);

00257   chThdSleepMilliseconds(50);           /* The thread just terminates.      */

00259 
00260   /* Clearing the zombie by scanning the registry.*/
00261   n3 = regscan();
