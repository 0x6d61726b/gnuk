00132 /**
00133  * @brief   Pads mode setup.
00134  * @details This function programs a pads group belonging to the same port
00135  *          with the specified mode.
00136  * @note    This function is not meant to be invoked directly by the
00137  *          application code.
00138  * @note    @p PAL_MODE_UNCONNECTED is implemented as push pull output at 2MHz.
00139  * @note    Writing on pads programmed as pull-up or pull-down has the side
00140  *          effect to modify the resistor setting because the output latched
00141  *          data is used for the resistor selection.
00142  *
00143  * @param[in] port      the port identifier
00144  * @param[in] mask      the group mask
00145  * @param[in] mode      the mode
00146  */
00147 void _pal_lld_setgroupmode(ioportid_t port,
00148                            ioportmask_t mask,
00149                            uint_fast8_t mode) {
00150   static const uint8_t cfgtab[] = {
00151     4,          /* PAL_MODE_RESET, implemented as input.*/
00152     2,          /* PAL_MODE_UNCONNECTED, implemented as push pull output 2MHz.*/
00153     4,          /* PAL_MODE_INPUT */
00154     8,          /* PAL_MODE_INPUT_PULLUP */
00155     8,          /* PAL_MODE_INPUT_PULLDOWN */
00156     0,          /* PAL_MODE_INPUT_ANALOG */
00157     3,          /* PAL_MODE_OUTPUT_PUSHPULL, 50MHz.*/
00158     7,          /* PAL_MODE_OUTPUT_OPENDRAIN, 50MHz.*/
00159     8,          /* Reserved.*/
00160     8,          /* Reserved.*/
00161     8,          /* Reserved.*/
00162     8,          /* Reserved.*/
00163     8,          /* Reserved.*/
00164     8,          /* Reserved.*/
00165     8,          /* Reserved.*/
00166     8,          /* Reserved.*/
00167     0xB,        /* PAL_MODE_STM32_ALTERNATE_PUSHPULL, 50MHz.*/
00168     0xF,        /* PAL_MODE_STM32_ALTERNATE_OPENDRAIN, 50MHz.*/
00169   };
00170   uint32_t mh, ml, crh, crl, cfg;
00171   unsigned i;
00172 
00173   if (mode == PAL_MODE_INPUT_PULLUP)
00174     port->BSRR = mask;
00175   else if (mode == PAL_MODE_INPUT_PULLDOWN)
00176     port->BRR = mask;
00177   cfg = cfgtab[mode];
00178   mh = ml = crh = crl = 0;
00179   for (i = 0; i < 8; i++) {
00180     ml <<= 4;
00181     mh <<= 4;
00182     crl <<= 4;
00183     crh <<= 4;
00184     if ((mask & 0x0080) == 0)
00185       ml |= 0xf;
00186     else
00187       crl |= cfg;
00188     if ((mask & 0x8000) == 0)
00189       mh |= 0xf;
00190     else
00191       crh |= cfg;
00192     mask <<= 1;
00193   }
00194   port->CRH = (port->CRH & mh) | crh;
00195   port->CRL = (port->CRL & ml) | crl;
00196 }
