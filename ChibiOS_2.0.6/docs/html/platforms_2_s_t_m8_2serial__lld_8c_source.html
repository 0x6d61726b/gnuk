00098 static void set_error(SerialDriver *sdp, uint8_t sr) {
00099   sdflags_t sts = 0;
00100 
00101   if (sr & 0x08)                            /* OR bit.                      */
00102     sts |= SD_OVERRUN_ERROR;
00103   if (sr & 0x04)                            /* NF bit.                      */
00104     sts |= SD_NOISE_ERROR;
00105   if (sr & 0x02)                            /* FE bit.                      */
00106     sts |= SD_FRAMING_ERROR;
00107   if (sr & 0x01)                            /* PE bit.                      */
00108     sts |= SD_PARITY_ERROR;
00109   chSysLockFromIsr();
00110   sdAddFlagsI(sdp, sts);
00111   chSysUnlockFromIsr();
00112 }

00120 /**
00121  * @brief   UART1 initialization.
00122  *
00123  * @param[in] config    architecture-dependent serial driver configuration
00124  */
00125 static void uart1_init(const SerialConfig *config) {
00126 
00127   UART1_BRR2 = ((uint8_t)(config->sc_brr >> 8) & (uint8_t)0xF0) |
00128                ((uint8_t)config->sc_brr & (uint8_t)0x0F);
00129   UART1_BRR1 = (uint8_t)(config->sc_brr >> 4);
00130   UART1_CR1  = config->sc_mode &
00131                SD_MODE_PARITY;              /* PIEN included.               */
00132   UART1_CR2  = 0x2C;                        /* RIEN | TEN | REN.            */
00133   UART1_CR3  = config->sc_mode & SD_MODE_STOP;
00134   UART1_CR4  = 0;
00135   UART1_CR5  = 0;
00136   UART1_PSCR = 1;
00137   (void)UART1_SR;
00138   (void)UART1_DR;
00139 }
