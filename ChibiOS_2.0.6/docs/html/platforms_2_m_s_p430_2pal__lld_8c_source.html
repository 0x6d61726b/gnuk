00110 /**
00111  * @brief Pads mode setup.
00112  * @details This function programs a pads group belonging to the same port
00113  *          with the specified mode.
00114  *
00115  * @param[in] port the port identifier
00116  * @param[in] mask the group mask
00117  * @param[in] mode the mode
00118  *
00119  * @note This function is not meant to be invoked directly by the application
00120  *       code.
00121  * @note @p PAL_MODE_UNCONNECTED is implemented as output as recommended by
00122  *       the MSP430x1xx Family User's Guide. Unconnected pads are set to
00123  *       high logic state by default.
00124  * @note This function does not alter the @p PxSEL registers. Alternate
00125  *       functions setup must be handled by device-specific code.
00126  */
00127 void _pal_lld_setgroupmode(ioportid_t port,
00128                            ioportmask_t mask,
00129                            uint_fast8_t mode) {
00130 
00131   switch (mode) {
00132   case PAL_MODE_RESET:
00133   case PAL_MODE_INPUT:
00134     port->iop_common.dir.reg_p &= ~mask;
00135     break;
00136   case PAL_MODE_UNCONNECTED:
00137     port->iop_common.out.reg_p |= mask;
00138   case PAL_MODE_OUTPUT_PUSHPULL:
00139     port->iop_common.dir.reg_p |= mask;
00140     break;
00141   }
00142 }
