00195 /**
00196  * @brief Writes a group of bits.
00197  * @details This function is implemented by writing the PIO_OWER, PIO_ODSR and
00198  *          PIO_OWDR registers, the implementation is not atomic because the
00199  *          multiple accesses.
00200  *
00201  * @param[in] port the port identifier
00202  * @param[in] mask the group mask
00203  * @param[in] offset the group bit offset within the port
00204  * @param[in] bits the bits to be written. Values exceeding the group width
00205  *            are masked.
00206  *
00207  * @note This function is not meant to be invoked directly by the application
00208  *       code.
00209  */
00210 #define pal_lld_writegroup(port, mask, offset, bits) {                  \
00211   (port)->PIO_OWER = (mask) << (offset);                                \
00212   (port)->PIO_ODSR = (bits) << (offset);                                \
00213   (port)->PIO_OWDR = (mask) << (offset);                                \
00214 }
