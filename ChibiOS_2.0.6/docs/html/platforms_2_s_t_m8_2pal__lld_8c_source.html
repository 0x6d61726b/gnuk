00060 /**
00061  * @brief   Pads mode setup.
00062  * @details This function programs a pads group belonging to the same port
00063  *          with the specified mode.
00064  * @note    This function is not meant to be invoked directly by the
00065  *          application code.
00066  * @note    @p PAL_MODE_UNCONNECTED is implemented as push pull output at 2MHz.
00067  *
00068  * @param[in] port      the port identifier
00069  * @param[in] mask      the group mask
00070  * @param[in] mode      the mode
00071  */
00072 void _pal_lld_setgroupmode(ioportid_t port,
00073                            ioportmask_t mask,
00074                            uint_fast8_t mode) {
00075 
00076   switch (mode & PAL_MODE_MASK) {
00077   case PAL_MODE_RESET:
00078   case PAL_MODE_INPUT_PULLUP:
00079     port->DDR &= ~mask;
00080     port->CR1 |=  mask;
00081     port->CR2 &= ~mask;
00082     break;
00083   case PAL_MODE_INPUT:
00084   case PAL_MODE_INPUT_ANALOG:
00085     port->DDR &= ~mask;
00086     port->CR1 &= ~mask;
00087     port->CR2 &= ~mask;
00088     break;
00089   case PAL_MODE_UNCONNECTED:
00090   case PAL_MODE_OUTPUT_PUSHPULL_SLOW:
00091     port->DDR |=  mask;
00092     port->CR1 |=  mask;
00093     port->CR2 &= ~mask;
00094     break;
00095   case PAL_MODE_OUTPUT_PUSHPULL:
00096     port->DDR |=  mask;
00097     port->CR1 |=  mask;
00098     port->CR2 |=  mask;
00099     break;
00100   case PAL_MODE_OUTPUT_OPENDRAIN_SLOW:
00101     port->DDR |=  mask;
00102     port->CR1 &= ~mask;
00103     port->CR2 &= ~mask;
00104     break;
00105   case PAL_MODE_OUTPUT_OPENDRAIN:
00106     port->DDR |=  mask;
00107     port->CR1 &= ~mask;
00108     port->CR2 |=  mask;
00109     break;
00110   }
00111 }
