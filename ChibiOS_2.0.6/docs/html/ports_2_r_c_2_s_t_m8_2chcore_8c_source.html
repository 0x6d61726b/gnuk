00041  * @brief   Performs a context switch between two threads.
00042  *
00043  * @param otp   the thread to be switched out
00044  */
00045 void _port_switch(Thread *otp) {
00046 
00047   (void)otp;
00048   /* Asm because unoptimal code would generated by using _getSP_().*/
00049 #pragma ASM
00050         LDW     Y,SP                ; old context pointer
00051         LDW     (5,X),Y             ; SP saved in otp->p_ctx.sp
00052         LDW     X,rlist + 5         ; r_current (currp) field
00053         LDW     X,(5,X)             ; currp->p_ctx.sp
00054         LDW     SP,X                ; new context pointer
00055 #pragma ENDASM
00056 }
