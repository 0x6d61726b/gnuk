00096 /**
00097  * @brief STM32 clocks and PLL initialization.
00098  * @note All the involved constants come from the file @p board.h.
00099  */
00100 #if defined(STM32F10X_LD) || defined(STM32F10X_MD) ||                       \
00101     defined(STM32F10X_HD) || defined(__DOXYGEN__)
00102 /*
00103  * Clocks initialization for the LD, MD and HD sub-families.
00104  */
00105 void stm32_clock_init(void) {
00106 
00107   /* HSI setup, it enforces the reset situation in order to handle possible
00108      problems with JTAG probes and re-initializations.*/
00109   RCC->CR |= RCC_CR_HSION;                  /* Make sure HSI is ON.         */
00110   while (!(RCC->CR & RCC_CR_HSIRDY))
00111     ;                                       /* Wait until HSI is stable.    */
00112   RCC->CR &= RCC_CR_HSITRIM | RCC_CR_HSION; /* CR Reset value.              */
00113   RCC->CFGR = 0;                            /* CFGR reset value.            */
00114   while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
00115     ;                                       /* Wait until HSI is the source.*/
00116 
00117   /* HSE setup, it is only performed if the HSE clock is selected as source
00118      of the system clock (directly or through the PLL).*/
00119 #if (STM32_SW == STM32_SW_HSE) ||                                           \
00120     ((STM32_SW == STM32_SW_PLL) && (STM32_PLLSRC == STM32_PLLSRC_HSE))
00121   RCC->CR |= RCC_CR_HSEON;
00122   while (!(RCC->CR & RCC_CR_HSERDY))
00123     ;                           /* Waits until HSE is stable.               */
00124 #endif
00125 
00126   /* PLL setup, it is only performed if the PLL is the selected source of
00127      the system clock else it is left disabled.*/
00128 #if STM32_SW == STM32_SW_PLL
00129   RCC->CFGR |= STM32_PLLMUL | STM32_PLLXTPRE | STM32_PLLSRC;
00130   RCC->CR   |= RCC_CR_PLLON;
00131   while (!(RCC->CR & RCC_CR_PLLRDY))
00132     ;                           /* Waits until PLL is stable.               */
00133 #endif
00134 
00135   /* Clock settings.*/
00136   RCC->CFGR = STM32_MCO | STM32_PLLMUL | STM32_PLLXTPRE | STM32_PLLSRC |
00137               STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 | STM32_HPRE;
00138 
00139   /* Flash setup and final clock selection.   */
00140   FLASH->ACR = STM32_FLASHBITS; /* Flash wait states depending on clock.    */
00141 
00142   /* Switching on the configured clock source if it is different from HSI.*/
00143 #if (STM32_SW != STM32_SW_HSI)
00144   RCC->CFGR |= STM32_SW;        /* Switches on the selected clock source.   */
00145   while ((RCC->CFGR & RCC_CFGR_SWS) != (STM32_SW << 2))
00146     ;
00147 #endif
00148 }

00174

00177 #if STM32_SW == STM32_SW_PLL
00178 #if STM32_PREDIV1SRC == STM32_PREDIV1SRC_PLL2
00179   RCC->CFGR2 |= STM32_PREDIV2 | STM32_PLL2MUL;
