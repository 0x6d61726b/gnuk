00078 static void set_error(uint8_t sra, SerialDriver *sdp) {
00079   sdflags_t sts = 0;
00080 
00081   if (sra & (1 << DOR))
00082     sts |= SD_OVERRUN_ERROR;
00083   if (sra & (1 << UPE))
00084     sts |= SD_PARITY_ERROR;
00085   if (sra & (1 << FE))
00086     sts |= SD_FRAMING_ERROR;
00087   chSysLockFromIsr();
00088   sdAddFlagsI(sdp, sts);
00089   chSysUnlockFromIsr();
00090 }
