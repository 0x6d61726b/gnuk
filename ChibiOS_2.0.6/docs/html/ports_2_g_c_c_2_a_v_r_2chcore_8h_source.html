00145  * @brief   Platform dependent part of the @p chThdInit() API.
00146  * @details This code usually setup the context switching frame represented
00147  *          by an @p intctx structure.
00148  */
00149 #define SETUP_CONTEXT(workspace, wsize, pf, arg) {                      \
00150   tp->p_ctx.sp = (struct intctx*)((uint8_t *)workspace + wsize  -       \
00151                                   sizeof(struct intctx));               \
00152   tp->p_ctx.sp->r2  = (int)pf;                                          \
00153   tp->p_ctx.sp->r3  = (int)pf >> 8;                                     \
00154   tp->p_ctx.sp->r4  = (int)arg;                                         \
