00246 /**
00247  * @brief   Writes a group of bits.
00248  * @details This function is implemented by writing the GPIO BSRR register, the
00249  *          implementation has no side effects.
00250  * @note    This function is not meant to be invoked directly by the
00251  *          application code.
00252  * @note    Writing on pads programmed as pull-up or pull-down has the side
00253  *          effect to modify the resistor setting because the output latched
00254  *          data is used for the resistor selection.
00255  *
00256  * @param[in] port      the port identifier
00257  * @param[in] mask      the group mask
00258  * @param[in] offset    the group bit offset within the port
00259  * @param[in] bits      the bits to be written. Values exceeding the group
00260  *                      width are masked.
00261  */
00262 #define pal_lld_writegroup(port, mask, offset, bits) {                      \
00263   (port)->BSRR = ((~(bits) & (mask)) << (16 + (offset))) |                  \
00264                  (((bits) & (mask)) << (offset));                           \
00265 }
