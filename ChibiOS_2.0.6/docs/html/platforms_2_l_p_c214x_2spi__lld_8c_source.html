00056 /**
00057  * @brief Synchronous SSP transfer.
00058  *
00059  * @param[in] n         number of bytes to be exchanged
00060  *
00061  * @param[in] txbuf     the pointer to the transmit buffer or @p NULL
00062  * @param[out] rxbuf    the pointer to the receive buffer or @p NULL
00063  */
00064 void rw8(size_t n, const uint8_t *txbuf, uint8_t *rxbuf) {
00065   size_t ntx = n;
00066 
00067   while (n > 0) {
00068     uint32_t sr = SSPBase->SSP_SR;
00069     if (sr & SR_RNE) {
00070       uint8_t w = SSPBase->SSP_DR;
00071       if (rxbuf != NULL)
00072         *rxbuf++ = w;
00073       n--;
00074       continue; /* Priority over transmission. */
00075     }
00076     if ((ntx > 0) && (sr & SR_TNF)) {
00077       if (txbuf != NULL)
00078         SSPBase->SSP_DR = *txbuf++;
00079       else
00080         SSPBase->SSP_DR = 0xFFFFFFFF;
00081       ntx--;
00082     }
00083   }
00084 }

00104 /**
00105  * @brief Configures and activates the SPI peripheral.
00106  *
00107  * @param[in] spip      pointer to the @p SPIDriver object
00108  */
00109 void spi_lld_start(SPIDriver *spip) {
00110 
00111   if (spip->spd_state == SPI_STOP) {
00112     /* Clock activation.*/
00113     PCONP = (PCONP & PCALL) | PCSPI1;
00114   }
00115   /* Configuration.*/
00116   SSPBase->SSP_CR1  = 0;
00117   /* Emptying the receive FIFO, it happens to not be empty while debugging.*/
00118   while (SSPBase->SSP_SR & SR_RNE)
00119     (void) SSPBase->SSP_DR;
00120   SSPBase->SSP_CR0  = spip->spd_config->spc_cr0;
00121   SSPBase->SSP_CPSR = spip->spd_config->spc_cpsr;
00122   SSPBase->SSP_CR1  = spip->spd_config->spc_cr1 | CR1_SSE;
00123 }

00130 void spi_lld_stop(SPIDriver *spip) {
00131 
00132   if (spip->spd_state != SPI_STOP) {
00133     SSPBase->SSP_CR1  = 0;
00134     SSPBase->SSP_CR0  = 0;
00135     SSPBase->SSP_CPSR = 0;

00137   }
