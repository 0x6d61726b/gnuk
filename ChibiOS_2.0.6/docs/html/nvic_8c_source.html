00038 /**
00039  * @brief Sets the priority of an interrupt handler and enables it.
00040  *
00041  * @param n the interrupt number
00042  * @param prio the interrupt priority mask
00043  *
00044  * @note The parameters are not tested for correctness.
00045  */
00046 void NVICEnableVector(uint32_t n, uint32_t prio) {
00047   unsigned sh = (n & 3) << 3;
00048 
00049   NVIC_IPR(n >> 2) = (NVIC_IPR(n >> 2) & ~(0xFF << sh)) | (prio << sh);
00050   NVIC_ICPR(n >> 5) = 1 << (n & 0x1F);
00051   NVIC_ISER(n >> 5) = 1 << (n & 0x1F);
00052 }
00053 
00054 /**
00055  * @brief Disables an interrupt handler.
00056  *
00057  * @param n the interrupt number
00058  *
00059  * @note The parameters are not tested for correctness.
00060  */
00061 void NVICDisableVector(uint32_t n) {
00062   unsigned sh = (n & 3) << 3;
00063 
00064   NVIC_ICER(n >> 5) = 1 << (n & 0x1F);
00065   NVIC_IPR(n >> 2) = NVIC_IPR(n >> 2) & ~(0xFF << sh);
00066 }

00068 /**
00069  * @brief Changes the priority of a system handler.
00070  *
00071  * @param handler the system handler number
00072  * @param prio the system handler priority mask
00073  * @note The parameters are not tested for correctness.
00074  */
00075 void NVICSetSystemHandlerPriority(uint32_t handler, uint32_t prio) {
00076   unsigned sh = (handler & 3) * 8;
00077 
00078   SCB_SHPR(handler >> 2) = (SCB_SHPR(handler >> 2) &
00079                            ~(0xFF << sh)) | (prio << sh);
00080 }
