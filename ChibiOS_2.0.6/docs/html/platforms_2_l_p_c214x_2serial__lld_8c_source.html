00072  * @param[in] sdp       communication channel associated to the UART
00073  * @param[in] config    the architecture-dependent serial driver configuration
00074  */
00075 static void uart_init(SerialDriver *sdp, const SerialConfig *config) {
00076   UART *u = sdp->uart;
00077 
00078   uint32_t div = PCLK / (config->sc_speed << 4);
00079   u->UART_LCR = config->sc_lcr | LCR_DLAB;
00080   u->UART_DLL = div;
00081   u->UART_DLM = div >> 8;
00082   u->UART_LCR = config->sc_lcr;
00083   u->UART_FCR = FCR_ENABLE | FCR_RXRESET | FCR_TXRESET | config->sc_fcr;
00084   u->UART_ACR = 0;
00085   u->UART_FDR = 0x10;
00086   u->UART_TER = TER_ENABLE;
00087   u->UART_IER = IER_RBR | IER_STATUS;

00133 /**
00134  * @brief   Common IRQ handler.
00135  * @note    Tries hard to clear all the pending interrupt sources, we dont want
00136  *          to go through the whole ISR and have another interrupt soon after.
00137  *
00138  * @param[in] sdp       communication channel associated to the UART
00139  */
00140 static void serve_interrupt(SerialDriver *sdp) {
00141   UART *u = sdp->uart;
00142 
00143   while (TRUE) {
00144     switch (u->UART_IIR & IIR_SRC_MASK) {
00145     case IIR_SRC_NONE:
00146       return;
00147     case IIR_SRC_ERROR:
00148       set_error(sdp, u->UART_LSR);
00149       break;
00150     case IIR_SRC_TIMEOUT:
00151     case IIR_SRC_RX:
00152       chSysLockFromIsr();
00153       if (chIQIsEmpty(&sdp->iqueue))
00154         chEvtBroadcastI(&sdp->ievent);
00155       chSysUnlockFromIsr();
00156       while (u->UART_LSR & LSR_RBR_FULL) {
00157         chSysLockFromIsr();
00158         if (chIQPutI(&sdp->iqueue, u->UART_RBR) < Q_OK)
00159            sdAddFlagsI(sdp, SD_OVERRUN_ERROR);
00160         chSysUnlockFromIsr();
00161       }
00162       break;
00163     case IIR_SRC_TX:
00164       {
00165         int i = LPC214x_UART_FIFO_PRELOAD;
00166         do {
00167           msg_t b;
00168 
00169           chSysLockFromIsr();
00170           b = chOQGetI(&sdp->oqueue);
00171           chSysUnlockFromIsr();
00172           if (b < Q_OK) {
00173             u->UART_IER &= ~IER_THRE;
00174             chSysLockFromIsr();
00175             chEvtBroadcastI(&sdp->oevent);
00176             chSysUnlockFromIsr();
00177             break;
00178           }
00179           u->UART_THR = b;
00180         } while (--i);
00181       }
00182       break;
00183     default:
00184       (void) u->UART_THR;
00185       (void) u->UART_RBR;
00186     }
00187   }
00188 }

00190 /**
00191  * @brief   Attempts a TX FIFO preload.
00192  */
00193 static void preload(SerialDriver *sdp) {
00194   UART *u = sdp->uart;
00195 
00196   if (u->UART_LSR & LSR_THRE) {
00197     int i = LPC214x_UART_FIFO_PRELOAD;
00198     do {
00199       msg_t b = chOQGetI(&sdp->oqueue);
00200       if (b < Q_OK) {
00201         chEvtBroadcastI(&sdp->oevent);
00202         return;
00203       }
00204       u->UART_THR = b;
00205     } while (--i);
00206   }
00207   u->UART_IER |= IER_THRE;
00208 }
