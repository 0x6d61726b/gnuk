00112 #if CH_DBG_ENABLE_ASSERTS
00113 /**
00114  * @brief   Condition assertion.
00115  * @details If the condition check fails then the kernel panics with the
00116  *          specified message and halts.
00117  * @note    The condition is tested only if the @p CH_DBG_ENABLE_ASSERTS switch
00118  *          is specified in @p chconf.h else the macro does nothing.
00119  * @note    The convention for the message is the following:<br>
00120  *          @<function_name@>(), #@<assert_number@>
00121  * @note    The remark string is not currently used except for putting a
00122  *          comment in the code about the assertion.
00123  *
00124  * @param[in] c         the condition to be verified to be true
00125  * @param[in] m         the text message
00126  * @param[in] r         a remark string
00127  */

00129   if (!(c))                                                             \
00130     chDbgPanic(m);                                                      \
00131 }
