00137 /*
00138  * Common exit point for all IRQ routines, it performs the rescheduling if
00139  * required.
00140  * System stack frame structure after a context switch in the
00141  * interrupt handler:
00142  *
00143  * High +------------+
00144  *      |   LR_USR   | -+
00145  *      |     R12    |  |
00146  *      |     R3     |  |
00147  *      |     R2     |  | External context: IRQ handler frame
00148  *      |     R1     |  |
00149  *      |     R0     |  |
00150  *      |     PC     |  |   (user code return address)
00151  *      |   PSR_USR  | -+   (user code status)
00152  *      |    ....    | <- mk_DoRescheduleI() stack frame, optimize it for space
00153  *      |     LR     | -+   (system code return address)
00154  *      |     R11    |  |
00155  *      |     R10    |  |
00156  *      |     R9     |  |
00157  *      |     R8     |  | Internal context: mk_SwitchI() frame
00158  *      |    (R7)    |  |   (optional, see CH_CURRP_REGISTER_CACHE)
00159  *      |     R6     |  |
00160  *      |     R5     |  |
00161  * SP-> |     R4     | -+
00162  * Low  +------------+
00163  */
00164 .balign 16
00165 #ifdef THUMB_NO_INTERWORKING
00166 .code 16
00167 .thumb_func
00168 .globl _port_irq_common
00169 _port_irq_common:
00170         bl      chSchIsRescRequiredExI
00171         mov     lr, pc
00172         bx      lr
00173 .code 32
00174 #else /* !THUMB_NO_INTERWORKING */
00175 .code 32
00176 .globl _port_irq_common
00177 _port_irq_common:
00178         bl      chSchIsRescRequiredExI
00179 #endif /* !THUMB_NO_INTERWORKING */
00180         cmp     r0, #0                          // Simply returns if a
00181         ldmeqfd sp!, {r0-r3, r12, lr}           // reschedule is not
00182         subeqs  pc, lr, #4                      // required.
00183 
00184         // Saves the IRQ mode registers in the system stack.
00185         ldmfd   sp!, {r0-r3, r12, lr}           // IRQ stack now empty.
00186         msr     CPSR_c, #MODE_SYS | I_BIT
00187         stmfd   sp!, {r0-r3, r12, lr}           // Registers on System Stack.
00188         msr     CPSR_c, #MODE_IRQ | I_BIT
00189         mrs     r0, SPSR
00190         mov     r1, lr
00191         msr     CPSR_c, #MODE_SYS | I_BIT
00192         stmfd   sp!, {r0, r1}                   // Push R0=SPSR, R1=LR_IRQ.
00193 
00194         // Context switch.
00195 #ifdef THUMB_NO_INTERWORKING
00196         add     r0, pc, #1
00197         bx      r0
00198 .code 16
00199         bl      chSchDoRescheduleI
00200         mov     lr, pc
00201         bx      lr
00202 .code 32
00203 #else /* !THUMB_NO_INTERWORKING */
00204         bl      chSchDoRescheduleI
00205 #endif /* !THUMB_NO_INTERWORKING */
00206 
00207         // Re-establish the IRQ conditions again.
00208         ldmfd   sp!, {r0, r1}                   // Pop R0=SPSR, R1=LR_IRQ.
00209         msr     CPSR_c, #MODE_IRQ | I_BIT
00210         msr     SPSR_fsxc, r0
00211         mov     lr, r1
00212         msr     CPSR_c, #MODE_SYS | I_BIT
00213         ldmfd   sp!, {r0-r3, r12, lr}
00214         msr     CPSR_c, #MODE_IRQ | I_BIT
00215         subs    pc, lr, #4
