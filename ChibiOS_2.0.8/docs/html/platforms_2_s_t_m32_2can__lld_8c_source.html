00063 CH_IRQ_HANDLER(Vector8C) {
00064 
00065   CH_IRQ_PROLOGUE();
00066 
00067   /* No more events until a message is transmitted.*/
00068   CAN1->TSR = CAN_TSR_RQCP0 | CAN_TSR_RQCP1 | CAN_TSR_RQCP2;
00069   chSysLockFromIsr();

00079  * CAN1 RX0 interrupt handler.
00080  */
00081 CH_IRQ_HANDLER(Vector90) {
00082   uint32_t rf0r;
00083 
00084   CH_IRQ_PROLOGUE();
00085 
00086   rf0r = CAN1->RF0R;
00087   if ((rf0r & CAN_RF0R_FMP0) > 0) {
00088     /* No more receive events until the queue 0 has been emptied.*/
00089     CAN1->IER &= ~CAN_IER_FMPIE0;

00092       chSemSignalI(&CAND1.cd_rxsem);

00094     chSysUnlockFromIsr();

00096   if ((rf0r & CAN_RF0R_FOVR0) > 0) {
00097     /* Overflow events handling.*/

00100     chSysLockFromIsr();
00101     chEvtBroadcastI(&CAND1.cd_error_event);

00121  * CAN1 SCE interrupt handler.
00122  */
00123 CH_IRQ_HANDLER(Vector98) {
00124   uint32_t msr;
00125 
00126   CH_IRQ_PROLOGUE();
00127 
00128   msr = CAN1->MSR;
00129   CAN1->MSR = CAN_MSR_ERRI | CAN_MSR_WKUI | CAN_MSR_SLAKI;
00130   /* Wakeup event.*/
00131   if (msr & CAN_MSR_WKUI) {
00132     chSysLockFromIsr();
00133     chEvtBroadcastI(&CAND1.cd_wakeup_event);
00134     chSysUnlockFromIsr();
00135   }
00136   /* Error event.*/
00137   if (msr & CAN_MSR_ERRI) {
00138     canstatus_t flags;
00139     uint32_t esr = CAN1->ESR;
00140 
00141     CAN1->ESR &= ~CAN_ESR_LEC;
00142     flags = (canstatus_t)(esr & 7);
00143     if ((esr & CAN_ESR_LEC) > 0)
00144       flags |= CAN_FRAMING_ERROR;
00145     chSysLockFromIsr();
00146     canAddFlagsI(&CAND1, flags);
00147     chEvtBroadcastI(&CAND1.cd_error_event);
00148     chSysUnlockFromIsr();
00149   }
00150

00176  *
00177  * @param[in] canp pointer to the @p CANDriver object
00178  */
00179 void can_lld_start(CANDriver *canp) {
00180 
00181   /* Clock activation.*/
00182 #if USE_STM32_CAN1
00183   if (&CAND1 == canp) {
00184     NVICEnableVector(USB_HP_CAN1_TX_IRQn,
00185                      CORTEX_PRIORITY_MASK(STM32_CAN1_IRQ_PRIORITY));
00186     NVICEnableVector(USB_LP_CAN1_RX0_IRQn,
00187                      CORTEX_PRIORITY_MASK(STM32_CAN1_IRQ_PRIORITY));

00192     RCC->APB1ENR |= RCC_APB1ENR_CAN1EN;
00193   }
00194 #endif
00195 
00196   /* Entering initialization mode. */
00197   canp->cd_state = CAN_STARTING;
00198   canp->cd_can->MCR = CAN_MCR_INRQ;
00199   while ((canp->cd_can->MSR & CAN_MSR_INAK) == 0)
00200     chThdSleepS(1);
00201   /* BTR initialization.*/
00202   canp->cd_can->BTR = canp->cd_config->cc_btr;
00203   /* MCR initialization.*/
00204   canp->cd_can->MCR = canp->cd_config->cc_mcr;
00205   /* Filters initialization.*/
00206   canp->cd_can->FMR |= CAN_FMR_FINIT;
00207   if (canp->cd_config->cc_num > 0) {
00208     uint32_t i, fmask;
00209     CAN_FilterRegister_TypeDef *cfp;
00210 
00211     canp->cd_can->FA1R = 0;
00212     canp->cd_can->FM1R = 0;
00213     canp->cd_can->FS1R = 0;
00214     canp->cd_can->FFA1R = 0;
00215     cfp = canp->cd_can->sFilterRegister;
00216     fmask = 1;
00217     for (i = 0; i < CAN_MAX_FILTERS; i++) {
00218       if (i < canp->cd_config->cc_num) {
00219         if (canp->cd_config->cc_filters[i].cf_mode)
00220           canp->cd_can->FM1R |= fmask;
00221         if (canp->cd_config->cc_filters[i].cf_scale)
00222           canp->cd_can->FS1R |= fmask;
00223         if (canp->cd_config->cc_filters[i].cf_assignment)
00224           canp->cd_can->FFA1R |= fmask;
00225         cfp->FR1 = canp->cd_config->cc_filters[i].cf_register1;
00226         cfp->FR2 = canp->cd_config->cc_filters[i].cf_register2;
00227         canp->cd_can->FA1R |= fmask;
00228       }
00229       else {
00230         cfp->FR1 = 0;
00231         cfp->FR2 = 0;
00232       }
00233       cfp++;
00234       fmask <<= 1;
00235       /* Gives a chance for preemption since this is a rather long loop.*/
00236       chSysUnlock();
00237       chThdYield();
00238       chSysLock();
00239     }
00240   }
00241   else {
00242     /* Setup a default filter.*/
00243     canp->cd_can->sFilterRegister[0].FR1 = 0;
00244     canp->cd_can->sFilterRegister[0].FR2 = 0;
00245     canp->cd_can->FM1R = 0;
00246     canp->cd_can->FFA1R = 0;
00247     canp->cd_can->FS1R = 1;
00248     canp->cd_can->FA1R = 1;
00249   }
00250   canp->cd_can->FMR &= ~CAN_FMR_FINIT;
00251   /* Interrupt sources initialization.*/
00252   canp->cd_can->IER = CAN_IER_TMEIE  | CAN_IER_FMPIE0 | CAN_IER_FMPIE1 |
00253                       CAN_IER_WKUIE  | CAN_IER_ERRIE  | CAN_IER_LECIE  |
00254                       CAN_IER_BOFIE  | CAN_IER_EPVIE  | CAN_IER_EWGIE  |
00255                       CAN_IER_FOVIE0 | CAN_IER_FOVIE1;
00256 }

00295 /**
00296  * @brief Inserts a frame into the transmit queue.
00297  *
00298  * @param[in] canp      pointer to the @p CANDriver object
00299  * @param[in] ctfp      pointer to the CAN frame to be transmitted
00300  */
00301 void can_lld_transmit(CANDriver *canp, const CANTxFrame *ctfp) {
00302   uint32_t tir;
00303   CAN_TxMailBox_TypeDef *tmbp;
00304 
00305   /* Pointer to a free transmission mailbox.*/
00306   tmbp = &canp->cd_can->sTxMailBox[(canp->cd_can->TSR & CAN_TSR_CODE) >> 24];
00307 
00308   /* Preparing the message.*/
00309   if (ctfp->cf_IDE)
00310     tir = ((uint32_t)ctfp->cf_EID << 3) | ((uint32_t)ctfp->cf_RTR << 1) |
00311           CAN_TI0R_IDE;
00312   else
00313     tir = ((uint32_t)ctfp->cf_SID << 21) | ((uint32_t)ctfp->cf_RTR << 1);
00314   tmbp->TDTR = ctfp->cf_DLC;
00315   tmbp->TDLR = ctfp->cf_data32[0];
00316   tmbp->TDHR = ctfp->cf_data32[1];
00317   tmbp->TIR = tir | CAN_TI0R_TXRQ;
00318 }

00334 /**
00335  * @brief Receives a frame from the input queue.
00336  *
00337  * @param[in] canp      pointer to the @p CANDriver object
00338  * @param[out] crfp     pointer to the buffer where the CAN frame is copied
00339  */
00340 void can_lld_receive(CANDriver *canp, CANRxFrame *crfp) {
00341   uint32_t r;
00342 
00343   /* Fetches the message.*/
00344   r = canp->cd_can->sFIFOMailBox[0].RIR;
00345   crfp->cf_RTR = (r & CAN_RI0R_RTR) >> 1;
00346   crfp->cf_IDE = (r & CAN_RI0R_IDE) >> 2;
00347   if (crfp->cf_IDE)
00348     crfp->cf_EID = r >> 3;
00349   else
00350     crfp->cf_SID = r >> 21;
00351   r = canp->cd_can->sFIFOMailBox[0].RDTR;
00352   crfp->cf_DLC = r & CAN_RDT0R_DLC;
00353   crfp->cf_FMI = (uint8_t)(r >> 8);
00354   crfp->cf_TIME = (uint16_t)(r >> 16);
00355   crfp->cf_data32[0] = canp->cd_can->sFIFOMailBox[0].RDLR;
00356   crfp->cf_data32[1] = canp->cd_can->sFIFOMailBox[0].RDHR;
00357 
00358   /* Releases the mailbox.*/
00359   canp->cd_can->RF0R = CAN_RF0R_RFOM0;
00360 
00361   /* If the queue is empty re-enables the interrupt in order to generate
00362      events again.*/
00363   if ((canp->cd_can->RF0R & CAN_RF0R_FMP0) == 0)
00364     canp->cd_can->IER |= CAN_IER_FMPIE0;
00365 }
