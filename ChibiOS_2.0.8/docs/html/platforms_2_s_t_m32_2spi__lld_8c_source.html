00064 static void spi_stop(SPIDriver *spip) {
00065 
00066   /* Stops RX and TX DMA channels.*/
00067   spip->spd_dmarx->CCR = 0;
00068   spip->spd_dmatx->CCR = 0;
00069 
00070   /* Stops SPI operations.*/
00071   spip->spd_spi->CR1 &= ~SPI_CR1_SPE;
00072 
00073   chSysLockFromIsr();
00074   chSchReadyI(spip->spd_thread);
00075   chSysUnlockFromIsr();
00076 }

00078 static void spi_start_wait(SPIDriver *spip, size_t n,
00079                            const void *txbuf, void *rxbuf) {
00080   uint32_t ccr;
00081 
00082   /* Common DMA setup.*/
00083   ccr = spip->spd_dmaprio;
00084   if ((spip->spd_config->spc_cr1 & SPI_CR1_DFF) != 0)
00085     ccr |= DMA_CCR1_MSIZE_0 | DMA_CCR1_PSIZE_0; /* 16 bits transfer.*/
00086 
00087   /* RX DMA setup.*/
00088   spip->spd_dmarx->CMAR = (uint32_t)rxbuf;
00089   spip->spd_dmarx->CNDTR = (uint32_t)n;
00090   spip->spd_dmarx->CCR |= ccr;
00091 
00092   /* TX DMA setup.*/
00093   spip->spd_dmatx->CMAR = (uint32_t)txbuf;
00094   spip->spd_dmatx->CNDTR = (uint32_t)n;
00095   spip->spd_dmatx->CCR |= ccr;
00096 
00097   /* SPI enable.*/
00098   chSysLock();
00099   spip->spd_spi->CR1 |= SPI_CR1_SPE;
00100 
00101   /* DMAs start.*/
00102   spip->spd_dmarx->CCR |= DMA_CCR1_EN;
00103   spip->spd_dmatx->CCR |= DMA_CCR1_EN;
00104 
00105   /* Wait for completion event.*/
00106   spip->spd_thread = currp;
00107   chSchGoSleepS(THD_STATE_SUSPENDED);
00108   spip->spd_thread = NULL;
00109   chSysUnlock();
00110 }

00216 /**
00217  * @brief Configures and activates the SPI peripheral.
00218  *
00219  * @param[in] spip      pointer to the @p SPIDriver object
00220  */
00221 void spi_lld_start(SPIDriver *spip) {
00222 
00223   /* If in stopped state then enables the SPI and DMA clocks.*/
00224   if (spip->spd_state == SPI_STOP) {
00225 #if USE_STM32_SPI1
00226     if (&SPID1 == spip) {
00227       dmaEnable(DMA1_ID);   /* NOTE: Must be enabled before the IRQs.*/
00228       NVICEnableVector(DMA1_Channel2_IRQn,
00229                        CORTEX_PRIORITY_MASK(STM32_SPI1_IRQ_PRIORITY));
00230       NVICEnableVector(DMA1_Channel3_IRQn,
00231                        CORTEX_PRIORITY_MASK(STM32_SPI1_IRQ_PRIORITY));
00232       RCC->APB2ENR |= RCC_APB2ENR_SPI1EN;
00233     }
00234 #endif
00235 #if USE_STM32_SPI2
00236     if (&SPID2 == spip) {
00237       dmaEnable(DMA1_ID);   /* NOTE: Must be enabled before the IRQs.*/
00238       NVICEnableVector(DMA1_Channel4_IRQn,
00239                        CORTEX_PRIORITY_MASK(STM32_SPI2_IRQ_PRIORITY));
00240       NVICEnableVector(DMA1_Channel5_IRQn,
00241                        CORTEX_PRIORITY_MASK(STM32_SPI2_IRQ_PRIORITY));
00242       RCC->APB1ENR |= RCC_APB1ENR_SPI2EN;
00243     }
00244 #endif
00245   }
00246 
00247   /* SPI setup.*/
00248   spip->spd_spi->CR1 = spip->spd_config->spc_cr1 | SPI_CR1_MSTR;
00249   spip->spd_spi->CR2 = SPI_CR2_SSOE | SPI_CR2_RXDMAEN | SPI_CR2_TXDMAEN;
00250 
00251   /* DMA setup.*/
00252   spip->spd_dmarx->CPAR = (uint32_t)&spip->spd_spi->DR;
00253   spip->spd_dmatx->CPAR = (uint32_t)&spip->spd_spi->DR;
00254 }

00311  * @param[in] spip      pointer to the @p SPIDriver object
