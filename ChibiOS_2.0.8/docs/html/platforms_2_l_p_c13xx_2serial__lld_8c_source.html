00070 static void uart_init(SerialDriver *sdp, const SerialConfig *config) {
00071   LPC_UART_TypeDef *u = sdp->uart;
00072 
00073   uint32_t div = LPC13xx_UART_PCLK / (config->sc_speed << 4);
00074   u->LCR = config->sc_lcr | LCR_DLAB;
00075   u->DLL = div;
00076   u->DLM = div >> 8;
00077   u->LCR = config->sc_lcr;
00078   u->FCR = FCR_ENABLE | FCR_RXRESET | FCR_TXRESET | config->sc_fcr;
00079   u->ACR = 0;
00080   u->FDR = 0x10;
00081   u->TER = TER_ENABLE;
00082   u->IER = IER_RBR | IER_STATUS;

00125 /**
00126  * @brief   Common IRQ handler.
00127  * @note    Tries hard to clear all the pending interrupt sources, we don't
00128  *          want to go through the whole ISR and have another interrupt soon
00129  *          after.
00130  *
00131  * @param[in] u         pointer to an UART I/O block
00132  * @param[in] sdp       communication channel associated to the UART
00133  */
00134 static void serve_interrupt(SerialDriver *sdp) {
00135   LPC_UART_TypeDef *u = sdp->uart;
00136 
00137   while (TRUE) {
00138     switch (u->IIR & IIR_SRC_MASK) {
00139     case IIR_SRC_NONE:
00140       return;
00141     case IIR_SRC_ERROR:
00142       set_error(sdp, u->LSR);
00143       break;
00144     case IIR_SRC_TIMEOUT:
00145     case IIR_SRC_RX:
00146       chSysLockFromIsr();
00147       if (chIQIsEmpty(&sdp->iqueue))
00148         chEvtBroadcastI(&sdp->ievent);
00149       chSysUnlockFromIsr();
00150       while (u->LSR & LSR_RBR_FULL) {
00151         chSysLockFromIsr();
00152         if (chIQPutI(&sdp->iqueue, u->RBR) < Q_OK)
00153            sdAddFlagsI(sdp, SD_OVERRUN_ERROR);
00154         chSysUnlockFromIsr();
00155       }
00156       break;
00157     case IIR_SRC_TX:
00158       {
00159         int i = LPC13xx_UART_FIFO_PRELOAD;
00160         do {
00161           msg_t b;
00162 
00163           chSysLockFromIsr();
00164           b = chOQGetI(&sdp->oqueue);
00165           chSysUnlockFromIsr();
00166           if (b < Q_OK) {
00167             u->IER &= ~IER_THRE;
00168             chSysLockFromIsr();
00169             chEvtBroadcastI(&sdp->oevent);
00170             chSysUnlockFromIsr();
00171             break;
00172           }
00173           u->THR = b;
00174         } while (--i);
00175       }
00176       break;
00177     default:
00178       (void) u->THR;
00179       (void) u->RBR;
00180     }
00181   }
00182 }

00184 /**
00185  * @brief   Attempts a TX FIFO preload.
00186  */
00187 static void preload(SerialDriver *sdp) {
00188   LPC_UART_TypeDef *u = sdp->uart;
00189 
00190   if (u->LSR & LSR_THRE) {
00191     int i = LPC13xx_UART_FIFO_PRELOAD;
00192     do {
00193       msg_t b = chOQGetI(&sdp->oqueue);
00194       if (b < Q_OK) {
00195         chEvtBroadcastI(&sdp->oevent);
00196         return;
00197       }
00198       u->THR = b;
00199     } while (--i);
00200   }
00201   u->IER |= IER_THRE;
00202 }
