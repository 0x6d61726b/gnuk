00075  * @brief   Low level HAL driver initialization.

00077 void hal_lld_init(void) {
00078 
00079   /* SysTick initialization using the system clock.*/
00080   NVICSetSystemHandlerPriority(HANDLER_SYSTICK, CORTEX_PRIORITY_SYSTICK);
00081   SysTick->LOAD = LPC13xx_SYSCLK / CH_FREQUENCY - 1;
00082   SysTick->VAL = 0;
00083   SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
00084                   SysTick_CTRL_ENABLE_Msk |
00085                   SysTick_CTRL_TICKINT_Msk;

00088 /**
00089  * @brief   LPC13xx clocks and PLL initialization.
00090  * @note    All the involved constants come from the file @p board.h.
00091  */
00092 void LPC13xx_clock_init(void) {
00093   unsigned i;
00094 
00095   /* Flash wait states setting, the code takes care to not touch TBD bits.*/
00096   FLASHCFG = (FLASHCFG & ~3) | LPC13xx_FLASHCFG_FLASHTIM;
00097 
00098   /* System oscillator initialization if required.*/
00099 #if LPC13xx_MAINCLK_SOURCE == SYSMAINCLKSEL_PLLOUT
00100 #if LPC13xx_PLLCLK_SOURCE == SYSPLLCLKSEL_SYSOSC
00101   LPC_SYSCON->SYSOSCCTRL = LPC13xx_SYSOSCCTRL;
00102   LPC_SYSCON->PDRUNCFG &= ~(1 << 5);            /* System oscillator ON.    */
00103   for (i = 0; i < 200; i++)
00104     __NOP();                                    /* Stabilization delay.     */
00105 #endif /* LPC13xx_PLLCLK_SOURCE == SYSPLLCLKSEL_SYSOSC */
00106 
00107   /* PLL initialization if required.*/
00108   LPC_SYSCON->SYSPLLCLKSEL = LPC13xx_PLLCLK_SOURCE;
00109   LPC_SYSCON->SYSPLLCLKUEN = 1;                 /* Really required?         */
00110   LPC_SYSCON->SYSPLLCLKUEN = 0;
00111   LPC_SYSCON->SYSPLLCLKUEN = 1;
00112   LPC_SYSCON->SYSPLLCTRL = LPC13xx_SYSPLLCTRL_MSEL | LPC13xx_SYSPLLCTRL_PSEL;
00113   LPC_SYSCON->PDRUNCFG &= ~(1 << 7);            /* System PLL ON.           */
00114   while ((LPC_SYSCON->SYSPLLSTAT & 1) == 0)     /* Wait PLL lock.           */
00115     ;
00116 #endif /* LPC13xx_MAINCLK_SOURCE == SYSMAINCLKSEL_PLLOUT */
00117 
00118   /* Main clock source selection.*/
00119   LPC_SYSCON->MAINCLKSEL = LPC13xx_MAINCLK_SOURCE;
00120   LPC_SYSCON->MAINCLKUEN = 1;                   /* Really required?         */
00121   LPC_SYSCON->MAINCLKUEN = 0;
00122   LPC_SYSCON->MAINCLKUEN = 1;
00123   while ((LPC_SYSCON->MAINCLKUEN & 1) == 0)     /* Wait switch completion.  */
00124     ;
00125 
00126   /* ABH divider initialization, peripheral clocks are initially disabled,
00127      the various device drivers will handle their own setup except GPIO and
00128      IOCON that are left enabled.*/
00129   LPC_SYSCON->SYSAHBCLKDIV = LPC13xx_SYSABHCLK_DIV;
00130   LPC_SYSCON->SYSAHBCLKCTRL = 0x0001005F;
00131 
00132   /* Peripheral clock dividers initialization.*/
00133   LPC_SYSCON->UARTCLKDIV = LPC13xx_UART_PCLK_DIV;
00134 
00135   /* Memory remapping, vectors always in ROM.*/
00136   LPC_SYSCON->SYSMEMREMAP = 2;
00137 }
