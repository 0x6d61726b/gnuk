00085 /**
00086  * @brief   Common TIM2...TIM4 IRQ handler.
00087  * @note    It is assumed that the various sources are only activated if the
00088  *          associated callback pointer is not equal to @p NULL in order to not
00089  *          perform an extra check in a potentially critical interrupt handler.
00090  */
00091 static void serve_interrupt(PWMDriver *pwmp) {
00092   uint16_t sr;
00093 
00094   sr = pwmp->pd_tim->SR & pwmp->pd_tim->DIER;
00095   pwmp->pd_tim->SR = ~(TIM_SR_CC1IF | TIM_SR_CC2IF | TIM_SR_CC3IF |
00096                        TIM_SR_CC4IF | TIM_SR_UIF);
00097   if ((sr & TIM_SR_CC1IF) != 0)
00098     pwmp->pd_config->pc_channels[0].pcc_callback(pwmp);
00099   if ((sr & TIM_SR_CC2IF) != 0)
00100     pwmp->pd_config->pc_channels[1].pcc_callback(pwmp);
00101   if ((sr & TIM_SR_CC3IF) != 0)
00102     pwmp->pd_config->pc_channels[2].pcc_callback(pwmp);
00103   if ((sr & TIM_SR_CC4IF) != 0)
00104     pwmp->pd_config->pc_channels[3].pcc_callback(pwmp);
00105   if ((sr & TIM_SR_UIF) != 0)
00106     pwmp->pd_config->pc_callback(pwmp);
00107 }
00108 #endif /* USE_STM32_PWM2 || USE_STM32_PWM3 || USE_STM32_PWM4 */

00256  * @param[in] pwmp      pointer to a @p PWMDriver object

00258 void pwm_lld_start(PWMDriver *pwmp) {
00259   uint16_t ccer;
00260 
00261   /* Reset channels.*/
00262   pwmp->pd_enabled_channels = 0;            /* All channels disabled.       */
00263 
00264   if (pwmp->pd_state == PWM_STOP) {
00265     /* Clock activation and timer reset.*/
00266 #if USE_STM32_PWM1
00267     if (&PWMD1 == pwmp) {
00268       RCC->APB2ENR |= RCC_APB2ENR_TIM1EN;
00269       RCC->APB2RSTR = RCC_APB2RSTR_TIM1RST;
00270       RCC->APB2RSTR = 0;
00271       NVICEnableVector(TIM1_UP_IRQn,
00272                        CORTEX_PRIORITY_MASK(STM32_PWM1_IRQ_PRIORITY));
00273       NVICEnableVector(TIM1_CC_IRQn,
00274                        CORTEX_PRIORITY_MASK(STM32_PWM1_IRQ_PRIORITY));
00275     }
00276 #endif
00277 #if USE_STM32_PWM2
00278     if (&PWMD2 == pwmp) {
00279       RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;

00281       RCC->APB1RSTR = 0;
00282       NVICEnableVector(TIM2_IRQn,
00283                        CORTEX_PRIORITY_MASK(STM32_PWM2_IRQ_PRIORITY));
00284     }
00285 #endif
00286 #if USE_STM32_PWM3
00287     if (&PWMD3 == pwmp) {
00288       RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
00289       RCC->APB1RSTR = RCC_APB1RSTR_TIM3RST;
00290       RCC->APB1RSTR = 0;
00291       NVICEnableVector(TIM3_IRQn,
00292                        CORTEX_PRIORITY_MASK(STM32_PWM3_IRQ_PRIORITY));
00293     }
00294 #endif
00295 #if USE_STM32_PWM4
00296     if (&PWMD4 == pwmp) {
00297       RCC->APB1ENR |= RCC_APB1ENR_TIM4EN;
00298       RCC->APB1RSTR = RCC_APB1RSTR_TIM4RST;
00299       RCC->APB1RSTR = 0;
00300       NVICEnableVector(TIM4_IRQn,
00301                        CORTEX_PRIORITY_MASK(STM32_PWM4_IRQ_PRIORITY));
00302     }
00303 #endif
00304   }
00305   else {
00306     /* Driver re-configuration scenario, it must be stopped first.*/
00307     /* Really required ?????????? */
00308     pwmp->pd_tim->CR1  = 0;                   /* Timer stopped.               */
00309     pwmp->pd_tim->CR2  = 0;                   /* Timer stopped.               */
00310     pwmp->pd_tim->SMCR = 0;                   /* Slave mode disabled.         */
00311     pwmp->pd_tim->CCR1 = 0;                   /* Comparator 1 disabled.       */
00312     pwmp->pd_tim->CCR2 = 0;                   /* Comparator 2 disabled.       */
00313     pwmp->pd_tim->CCR3 = 0;                   /* Comparator 3 disabled.       */
00314     pwmp->pd_tim->CCR4 = 0;                   /* Comparator 4 disabled.       */
00315     pwmp->pd_tim->CNT  = 0;
00316   }
00317 
00318   /* Timer configuration.*/
00319   pwmp->pd_tim->CR2  = pwmp->pd_config->pc_cr2;
00320   pwmp->pd_tim->PSC  = pwmp->pd_config->pc_psc;
00321   pwmp->pd_tim->ARR  = pwmp->pd_config->pc_arr;
00322   /* Output enables and polarities setup.*/
00323   ccer = 0;
00324   switch (pwmp->pd_config->pc_channels[0].pcc_mode) {
00325   case PWM_OUTPUT_ACTIVE_LOW:

00333   case PWM_OUTPUT_ACTIVE_LOW:

00335   case PWM_OUTPUT_ACTIVE_HIGH:
00336     ccer |= TIM_CCER_CC2E;
00337   default:
00338     ;
00339   }
00340   switch (pwmp->pd_config->pc_channels[2].pcc_mode) {
00341   case PWM_OUTPUT_ACTIVE_LOW:
00342     ccer |= TIM_CCER_CC3P;
00343   case PWM_OUTPUT_ACTIVE_HIGH:
00344     ccer |= TIM_CCER_CC3E;
00345   default:
00346     ;
00347   }

00350     ccer |= TIM_CCER_CC4P;

00352     ccer |= TIM_CCER_CC4E;
00353   default:
00354     ;
00355   }
00356   pwmp->pd_tim->CCER = ccer;
00357   pwmp->pd_tim->EGR  = TIM_EGR_UG;          /* Update event.                */
00358   pwmp->pd_tim->SR   = 0;                   /* Clear pending IRQs.          */
00359   pwmp->pd_tim->DIER = pwmp->pd_config->pc_callback == NULL ? 0 : TIM_DIER_UIE;

00414 /**
00415  * @brief   Enables a PWM channel.
00416  *
00417  * @param[in] pwmp      pointer to a @p PWMDriver object
00418  * @param[in] channel   PWM channel identifier (0...PWM_CHANNELS-1)
00419  * @param[in] width     PWM pulse width as clock pulses number
00420  */
00421 void pwm_lld_enable_channel(PWMDriver *pwmp,
00422                             pwmchannel_t channel,
00423                             pwmcnt_t width) {
00424 
00425   *(&pwmp->pd_tim->CCR1 + (channel * 2)) = width;   /* New duty cycle.      */
00426   if ((pwmp->pd_enabled_channels & (1 << channel)) == 0) {
00427     /* The channel is not enabled yet.*/
00428     pwmp->pd_enabled_channels |= (1 << channel);
00429     /* If there is a callback associated to the channel then the proper
00430        interrupt is cleared and enabled.*/
00431     if (pwmp->pd_config->pc_channels[channel].pcc_callback) {
00432       pwmp->pd_tim->SR    = ~(2 << channel);
00433       pwmp->pd_tim->DIER |= (2 << channel);
00434     }
00435   }
00436 }

00438 /**
00439  * @brief   Disables a PWM channel.
00440  * @details The channel is disabled and its output line returned to the
00441  *          idle state.
00442  *
00443  * @param[in] pwmp      pointer to a @p PWMDriver object
00444  * @param[in] channel   PWM channel identifier (0...PWM_CHANNELS-1)
00445  */
00446 void pwm_lld_disable_channel(PWMDriver *pwmp, pwmchannel_t channel) {
00447 
00448   *(&pwmp->pd_tim->CCR1 + (channel * 2)) = 0;
00449   pwmp->pd_tim->DIER &= ~(2 << channel);
00450   pwmp->pd_enabled_channels &= ~(1 << channel);
