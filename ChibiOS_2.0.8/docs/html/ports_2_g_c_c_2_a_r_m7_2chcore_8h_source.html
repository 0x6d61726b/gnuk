00182 /**
00183  * @brief   IRQ prologue code.
00184  * @details This macro must be inserted at the start of all IRQ handlers
00185  *          enabled to invoke system APIs.
00186  * @note    This macro has a different implementation depending if compiled in
00187  *          ARM or THUMB mode.
00188  * @note    The THUMB implementation starts with ARM code because interrupt
00189  *          vectors are always invoked in ARM mode regardless the bit 0
00190  *          value. The switch in THUMB mode is done in the function prologue so

00194 #define PORT_IRQ_PROLOGUE() {                                               \
00195   asm volatile (".code 32                               \n\t"               \
00196                 "stmfd   sp!, {r0-r3, r12, lr}          \n\t"               \
00197                 "add     r0, pc, #1                     \n\t"               \
00198                 "bx      r0                             \n\t"               \
00199                 ".code 16" : : : "memory");                                 \
00200 }

00203   asm volatile ("stmfd    sp!, {r0-r3, r12, lr}" : : : "memory");           \

00340 /**
00341  * @brief   Performs a context switch between two threads.
00342  * @details This is the most critical code in any port, this function
00343  *          is responsible for the context switch between 2 threads.
00344  * @note    The implementation of this code affects <b>directly</b> the context
00345  *          switch performance so optimize here as much as you can.
00346  * @note    Implemented as inlined code for performance reasons.
00347  *
00348  * @param[in] ntp       the thread to be switched in
00349  * @param[in] otp       the thread to be switched out
00350  */
00351 #ifdef THUMB
00352 #if CH_DBG_ENABLE_STACK_CHECK
00353 #define port_switch(ntp, otp) {                                             \
00354   register Thread *_ntp asm ("r0") = (ntp);                                 \
00355   register Thread *_otp asm ("r1") = (otp);                                 \
00356   register char *sp asm ("sp");                                             \
00357   if (sp - sizeof(struct intctx) - sizeof(Thread) < (char *)_otp)           \
00358     asm volatile ("mov     r0, #0                               \n\t"       \
00359                   "ldr     r1, =chDbgPanic                      \n\t"       \
00360                   "bx      r1");                                            \
00361     _port_switch_thumb(_ntp, _otp);                                         \
00362 }
00363 #else /* !CH_DBG_ENABLE_STACK_CHECK */
00364 #define port_switch(ntp, otp) _port_switch_thumb(ntp, otp)
00365 #endif /* !CH_DBG_ENABLE_STACK_CHECK */
00366 #else /* !THUMB */
