00037 /**
00038  * @brief   Performs a context switch between two threads.
00039  * @details This is the most critical code in any port, this function
00040  *          is responsible for the context switch between 2 threads.
00041  * @note    The implementation of this code affects <b>directly</b> the context
00042  *          switch performance so optimize here as much as you can.
00043  * @note    The function is declared as a weak symbol, it is possible to
00044  *          redefine it in your application code.
00045  *
00046  * @param[in] ntp       the thread to be switched in
00047  * @param[in] otp       the thread to be switched out
00048  */
00049 #if !defined(__DOXYGEN__)
00050 __attribute__((naked, weak))
00051 #endif
00052 void port_switch(Thread *ntp, Thread *otp) {
00053   register struct intctx *sp asm("r1");
00054 
00055   asm volatile ("push    r11                                    \n\t" \
00056                 "push    r10                                    \n\t" \
00057                 "push    r9                                     \n\t" \
00058                 "push    r8                                     \n\t" \
00059                 "push    r7                                     \n\t" \
00060                 "push    r6                                     \n\t" \
00061                 "push    r5                                     \n\t" \
00062                 "push    r4" : : : "memory");
00063   otp->p_ctx.sp = sp;
00064   sp = ntp->p_ctx.sp;
00065   asm volatile ("pop     r4                                     \n\t" \
00066                 "pop     r5                                     \n\t" \
00067                 "pop     r6                                     \n\t" \
00068                 "pop     r7                                     \n\t" \
00069                 "pop     r8                                     \n\t" \
00070                 "pop     r9                                     \n\t" \
00071                 "pop     r10                                    \n\t" \
00072                 "pop     r11                                    \n\t" \
00073                 "ret" : : "r" (sp) : "memory");
00074 }
