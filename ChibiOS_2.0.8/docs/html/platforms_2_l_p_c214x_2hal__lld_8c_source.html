00094 /**
00095  * @brief LPC214x clocks and PLL initialization.
00096  * @note All the involved constants come from the file @p board.h.
00097  */
00098 void lpc214x_clock_init(void) {
00099 
00100   /*
00101    * All peripherals clock disabled by default in order to save power.
00102    */
00103   PCONP = PCRTC | PCTIM0;
00104 
00105   /*
00106    * MAM setup.
00107    */
00108   MAMTIM = 0x3;                 /* 3 cycles for flash accesses. */
00109   MAMCR  = 0x2;                 /* MAM fully enabled. */
00110 
00111   /*
00112    * PLL setup for Fosc=12MHz and CCLK=48MHz.
00113    * P=2 M=3.
00114    */
00115   PLL *pll = PLL0Base;
00116   pll->PLL_CFG  = 0x23;         /* P and M values. */
00117   pll->PLL_CON  = 0x1;          /* Enables the PLL 0. */
00118   pll->PLL_FEED = 0xAA;
00119   pll->PLL_FEED = 0x55;
00120   while (!(pll->PLL_STAT & 0x400))
00121     ;                           /* Wait for PLL lock. */
00122 
00123   pll->PLL_CON  = 0x3;          /* Connects the PLL. */
00124   pll->PLL_FEED = 0xAA;
00125   pll->PLL_FEED = 0x55;
00126 
00127   /*
00128    * VPB setup.
00129    * PCLK = CCLK / 4.
00130    */
00131   VPBDIV = VPD_D4;
00132 }
