00226 /**
00227  * @brief   PWM clock prescaler initialization utility.
00228  * @note    The real clock value is rounded to the lower valid value, please
00229  *          make sure that the source clock frequency is a multiple of the
00230  *          requested PWM clock frequency.
00231  * @note    The calculated value must fit into an unsigned 16 bits integer.
00232  *
00233  * @param[in] clksrc    clock source frequency, depending on the target timer
00234  *                      cell it can be one of:
00235  *                      - STM32_TIMCLK1
00236  *                      - STM32_TIMCLK2
00237  *                      .
00238  *                      Please refer to the STM32 HAL driver documentation
00239  *                      and/or the STM32 Reference Manual for the right clock
00240  *                      source.
00241  * @param[in] pwmclk    PWM clock frequency in cycles
00242  * @return              The value to be stored in the @p pc_psc field of the
00243  *                      @p PWMConfig structure.
00244  */
00245 #define PWM_COMPUTE_PSC(clksrc, pwmclk)                                     \
00246   ((uint16_t)(((clksrc) / (pwmclk)) - 1))
00247 
00248 /**
00249  * @brief   PWM cycle period initialization utility.
00250  * @note    The calculated value must fit into an unsigned 16 bits integer.
00251  *
00252  * @param[in] pwmclk    PWM clock frequency in cycles
00253  * @param[in] pwmperiod PWM cycle period in nanoseconds
00254  * @return              The value to be stored in the @p pc_arr field of the
00255  *                      @p PWMConfig structure.
00256  */
00257 #define PWM_COMPUTE_ARR(pwmclk, pwmperiod)                                  \
00258   ((uint16_t)(((pwmclk) / (1000000000 / (pwmperiod))) - 1))

00260 /**
00261  * @brief   Converts from fraction to pulse width.
00262  * @note    Be careful with rounding errors, this is integer math not magic.
00263  *          You can specify tenths of thousandth but make sure you have the
00264  *          proper hardware resolution by carefully choosing the clock source
00265  *          and prescaler settings, see @p PWM_COMPUTE_PSC.
00266  *
00267  * @param[in] numerator numerator of the fraction
00268  * @param[in] denominator percentage as an integer between 0 and numerator
00269  * @return              The pulse width to be passed to @p pwmEnableChannel().
00270  *
00271  * @api
00272  */
00273 #define PWM_FRACTION_TO_WIDTH(pwmp, numerator, denominator)                 \
00274   ((uint16_t)((((uint32_t)(pwmp)->pd_config->pc_arr + 1UL) *                \
00275                (uint32_t)(denominator)) / (uint32_t)(numerator)))
